1//! `RustyNES` Desktop - NES Emulator Frontend
2//!
3//! A desktop frontend for the `RustyNES` emulator using egui + eframe.
4//!
5//! # Usage
6//!
7//! ```bash
8//! rustynes [OPTIONS] [ROM_PATH]
9//! ```
10//!
11//! # Examples
12//!
13//! ```bash
14//! # Launch without a ROM (opens file dialog)
15//! rustynes
16//!
17//! # Launch with a specific ROM
18//! rustynes path/to/game.nes
19//!
20//! # Launch with debug mode enabled
21//! rustynes --debug path/to/game.nes
22//! ```
23
24#![cfg_attr(not(debug_assertions), windows_subsystem = "windows")]
25
26use anyhow::{Context, Result};
27use clap::Parser;
28use log::{error, info};
29use std::path::PathBuf;
30
31use rustynes_desktop::{Config, NesApp};
32
33/// Command-line arguments for `RustyNES`
34#[derive(Parser, Debug)]
35#[command(name = "rustynes")]
36#[command(author = "doublegate")]
37#[command(version = "0.8.6")]
38#[command(about = "A cycle-accurate NES emulator written in Rust")]
39#[command(long_about = None)]
40struct Args {
41    /// Path to a NES ROM file (.nes)
42    #[arg(value_name = "ROM")]
43    rom_path: Option<PathBuf>,
44
45    /// Start in fullscreen mode
46    #[arg(short, long)]
47    fullscreen: bool,
48
49    /// Window scale factor (1-8)
50    #[arg(short, long, default_value = "3")]
51    scale: u32,
52
53    /// Enable debug mode (shows debug windows)
54    #[arg(short, long)]
55    debug: bool,
56
57    /// Mute audio on startup
58    #[arg(short, long)]
59    mute: bool,
60}
61
62fn main() -> Result<()> {
63    // Initialize logging
64    // Filter calloop warnings: these are benign race conditions from the Wayland
65    // backend where key repeat events arrive after the event source is deregistered.
66    // See: https://github.com/bevyengine/bevy/issues/14904
67    env_logger::Builder::from_env(env_logger::Env::default().default_filter_or("info"))
68        .filter_module("calloop", log::LevelFilter::Error)
69        .format_timestamp_millis()
70        .init();
71
72    info!("RustyNES v0.8.6 starting...");
73
74    // Parse command-line arguments
75    let args = Args::parse();
76
77    // Load or create configuration
78    let mut config = Config::load().unwrap_or_else(|e| {
79        error!("Failed to load config, using defaults: {e}");
80        Config::default()
81    });
82
83    // Apply command-line overrides
84    if args.fullscreen {
85        config.video.fullscreen = true;
86    }
87    if args.scale >= 1 && args.scale <= 8 {
88        config.video.scale = args.scale;
89    }
90    if args.debug {
91        config.debug.enabled = true;
92    }
93    if args.mute {
94        config.audio.muted = true;
95    }
96
97    // Calculate initial window size (scale is bounded 1-8, safe for f32)
98    #[allow(clippy::cast_precision_loss)]
99    let (width, height) = {
100        let scale = f32::from(config.video.scale as u16);
101        (256.0 * scale, 240.0 * scale + 25.0) // Extra for menu bar
102    };
103
104    // Set up native options for eframe
105    let native_options = eframe::NativeOptions {
106        viewport: egui::ViewportBuilder::default()
107            .with_inner_size([width, height])
108            .with_min_inner_size([256.0, 265.0])
109            .with_title("RustyNES")
110            .with_fullscreen(config.video.fullscreen),
111        vsync: config.video.vsync,
112        ..Default::default()
113    };
114
115    // Create and run the application
116    eframe::run_native(
117        "RustyNES",
118        native_options,
119        Box::new(move |cc| Ok(Box::new(NesApp::new(cc, config, args.rom_path)))),
120    )
121    .map_err(|e| anyhow::anyhow!("eframe error: {e}"))
122    .context("Failed to run application")?;
123
124    Ok(())
125}
