1//! NES memory bus implementation.
2//!
3//! This module implements the NES bus which routes CPU memory access to various
4//! components: internal RAM, PPU registers, APU registers, controllers, and cartridge.
5//!
6//! # Memory Map
7//!
8//! ```text
9//! $0000-$07FF: 2KB internal RAM
10//! $0800-$1FFF: Mirrors of $0000-$07FF (3×)
11//! $2000-$2007: PPU registers
12//! $2008-$3FFF: Mirrors of $2000-$2007
13//! $4000-$4013: APU and I/O registers
14//! $4014:       OAM DMA register
15//! $4015:       APU status register
16//! $4016:       Controller 1 data / strobe
17//! $4017:       Controller 2 data / APU frame counter
18//! $4018-$401F: APU/I/O test mode registers (normally disabled)
19//! $4020-$FFFF: Cartridge space (mapper-controlled)
20//! ```
21

28/// NES memory bus connecting CPU to all components
29pub struct Bus {
30    /// 2KB internal RAM
31    ram: [u8; 0x800],
32
33    /// 8KB PRG-RAM / battery-backed SRAM ($6000-$7FFF)
34    /// Used by many test ROMs to report results
35    prg_ram: [u8; 0x2000],
36
37    /// Picture Processing Unit
38    pub ppu: Ppu,
39
40    /// Audio Processing Unit
41    pub apu: Apu,
42
43    /// Cartridge mapper
44    pub mapper: Box<dyn Mapper>,
45
46    /// Controller 1
47    pub controller1: Controller,
48
49    /// Controller 2
50    pub controller2: Controller,
51
52    /// OAM DMA state
53    dma_page: u8,
54    dma_addr: u8,
55    dma_data: u8,
56    dma_dummy_cycles: u8, // Number of dummy cycles remaining (1 or 2 depending on alignment)
57    dma_transfer: bool,
58    dma_write: bool,
59
60    /// CPU cycle counter for odd/even tracking (for DMA timing)
61    cpu_cycles: u64,
62}

64impl Bus {
65    /// Create a new bus with the given mapper
66    ///
67    /// # Arguments
68    ///
69    /// * `mapper` - Cartridge mapper implementation
70    /// * `mirroring` - Initial nametable mirroring mode
71    ///
72    /// # Returns
73    ///
74    /// New Bus instance with all components initialized
75    #[must_use]
76    pub fn new(mapper: Box<dyn Mapper>) -> Self {
77        let mirroring = mapper.mirroring();
78
79        // Convert mapper Mirroring to PPU Mirroring
80        let ppu_mirroring = match mirroring {
81            rustynes_mappers::Mirroring::Horizontal => PpuMirroring::Horizontal,
82            rustynes_mappers::Mirroring::Vertical => PpuMirroring::Vertical,
83            rustynes_mappers::Mirroring::SingleScreenLower => PpuMirroring::SingleScreenLower,
84            rustynes_mappers::Mirroring::SingleScreenUpper => PpuMirroring::SingleScreenUpper,
85            rustynes_mappers::Mirroring::FourScreen => PpuMirroring::FourScreen,
86        };
87
88        Self {
89            ram: [0; 0x800],
90            prg_ram: [0; 0x2000], // Initialize PRG-RAM to 0 (matches internal RAM)
91            ppu: Ppu::new(ppu_mirroring),
92            apu: Apu::new(),
93            mapper,
94            controller1: Controller::new(),
95            controller2: Controller::new(),
96            dma_page: 0,
97            dma_addr: 0,
98            dma_data: 0,
99            dma_dummy_cycles: 0,
100            dma_transfer: false,
101            dma_write: false,
102            cpu_cycles: 0,
103        }
104    }
105
106    /// Check if DMA transfer is active
107    ///
108    /// # Returns
109    ///
110    /// true if OAM DMA is in progress, false otherwise
111    #[must_use]
112    pub fn dma_active(&self) -> bool {
113        self.dma_transfer
114    }
115
116    /// Execute one cycle of DMA transfer
117    ///
118    /// OAM DMA takes 513 or 514 CPU cycles:
119    /// - 513 cycles: Started on even CPU cycle (1 dummy + 512 transfer cycles)
120    /// - 514 cycles: Started on odd CPU cycle (2 dummy cycles for alignment + 512 transfer cycles)
121    /// - Transfer: 256 alternating read/write cycles (512 total)
122    ///
123    /// # Returns
124    ///
125    /// true if DMA is complete, false if still in progress
126    pub fn tick_dma(&mut self) -> bool {
127        if !self.dma_transfer {
128            return true;
129        }
130
131        // Dummy cycle(s) for alignment
132        // If started on odd cycle, we need 2 dummy cycles
133        // If started on even cycle, we need 1 dummy cycle
134        if self.dma_dummy_cycles > 0 {
135            self.dma_dummy_cycles -= 1;
136            self.cpu_cycles += 1;
137            return false;
138        }
139
140        if self.dma_write {
141            // Write cycle
142            self.ppu.write_register(0x2004, self.dma_data, |_, _| {});
143            self.dma_write = false;
144            self.cpu_cycles += 1;
145
146            // Advance to next byte
147            self.dma_addr = self.dma_addr.wrapping_add(1);
148
149            // Check if transfer is complete (256 bytes transferred)
150            if self.dma_addr == 0 {
151                self.dma_transfer = false;
152                true
153            } else {
154                false
155            }
156        } else {
157            // Read cycle
158            let addr = u16::from(self.dma_page) << 8 | u16::from(self.dma_addr);
159            self.dma_data = self.read_for_dma(addr);
160            self.dma_write = true;
161            self.cpu_cycles += 1;
162            false
163        }
164    }
165
166    /// Read memory for DMA (doesn't trigger side effects)
167    ///
168    /// This is separate from regular `read()` to avoid triggering PPU/APU side effects
169    /// during DMA operations.
170    fn read_for_dma(&self, addr: u16) -> u8 {
171        match addr {
172            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize],
173            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize],
174            0x4020..=0xFFFF => self.mapper.read_prg(addr),
175            _ => 0, // DMA from PPU/APU registers returns open bus
176        }
177    }
178
179    /// Initiate OAM DMA transfer
180    ///
181    /// # Arguments
182    ///
183    /// * `page` - High byte of source address (e.g., $02 for $0200-$02FF)
184    ///
185    /// # Timing
186    ///
187    /// OAM DMA takes 513 or 514 CPU cycles:
188    /// - 513 cycles if started on an even CPU cycle (1 dummy + 512 transfer)
189    /// - 514 cycles if started on an odd CPU cycle (2 dummy + 512 transfer)
190    ///
191    /// The extra cycle on odd alignment ensures reads happen on even cycles.
192    fn start_oam_dma(&mut self, page: u8) {
193        self.dma_page = page;
194        self.dma_addr = 0;
195        self.dma_transfer = true;
196        self.dma_write = false;
197
198        // Determine number of dummy cycles based on CPU cycle parity
199        // Odd cycle: need 2 dummy cycles for alignment (514 total)
200        // Even cycle: need 1 dummy cycle (513 total)
201        self.dma_dummy_cycles = if (self.cpu_cycles % 2) == 1 { 2 } else { 1 };
202    }
203
204    /// Increment CPU cycle counter (called from console after each CPU instruction)
205    ///
206    /// This tracks odd/even cycles for DMA timing precision.
207    ///
208    /// # Arguments
209    ///
210    /// * `cycles` - Number of CPU cycles to add
211    pub fn add_cpu_cycles(&mut self, cycles: u8) {
212        self.cpu_cycles += u64::from(cycles);
213    }
214
215    /// Reset the bus and all components
216    pub fn reset(&mut self) {
217        self.ram = [0; 0x800];
218        self.prg_ram.fill(0); // Initialize PRG-RAM to 0
219        self.ppu.reset();
220        self.apu.reset();
221        self.controller1.reset();
222        self.controller2.reset();
223        self.dma_transfer = false;
224        self.cpu_cycles = 0;
225    }
226
227    /// Clock the mapper (for IRQ timing)
228    ///
229    /// # Arguments
230    ///
231    /// * `cycles` - Number of CPU cycles elapsed
232    pub fn clock_mapper(&mut self, cycles: u8) {
233        self.mapper.clock(cycles);
234    }
235
236    /// Notify mapper of PPU A12 rising edge (for scanline IRQ)
237    pub fn ppu_a12_edge(&mut self) {
238        self.mapper.ppu_a12_edge();
239    }

241    /// Check if mapper IRQ is pending
242    ///
243    /// # Returns
244    ///
245    /// true if mapper is asserting IRQ line
246    #[must_use]
247    pub fn mapper_irq_pending(&self) -> bool {
248        self.mapper.irq_pending()
249    }
250
251    /// Clear mapper IRQ
252    pub fn clear_mapper_irq(&mut self) {
253        self.mapper.clear_irq();
254    }
255
256    /// Step PPU by one dot with CHR ROM access from mapper
257    ///
258    /// This method bridges the PPU and mapper, allowing the PPU to fetch
259    /// pattern table data (CHR ROM) during background and sprite rendering.
260    ///
261    /// # Returns
262    ///
263    /// Tuple of (`frame_complete`, `nmi`):
264    /// - `frame_complete`: true if a complete frame was just rendered
265    /// - `nmi`: true if NMI should be triggered (`VBlank` start with NMI enabled)
266    pub fn step_ppu(&mut self) -> (bool, bool) {
267        // Borrow mapper immutably for CHR reads while PPU is borrowed mutably
268        // This works because they are separate fields of the struct
269        let mapper = &*self.mapper;
270        self.ppu.step_with_chr(|addr| mapper.read_chr(addr))
271    }
272}

274impl CpuBus for Bus {
275    fn read(&mut self, addr: u16) -> u8 {
276        match addr {
277            // 2KB internal RAM, mirrored 4 times
278            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize],
279
280            // PPU registers, mirrored every 8 bytes
281            0x2000..=0x3FFF => {
282                let ppu_addr = 0x2000 + (addr & 0x0007);
283                // Pass closure to read CHR from mapper
284                let mapper = &*self.mapper;
285                self.ppu
286                    .read_register(ppu_addr, |chr_addr| mapper.read_chr(chr_addr))
287            }
288
289            // APU and I/O registers
290            0x4000..=0x4015 => self.apu.read_register(addr),
291
292            // Controller 1
293            0x4016 => self.controller1.read(),
294
295            // Controller 2 (note: $4017 write goes to APU, read goes to controller)
296            0x4017 => self.controller2.read(),
297
298            // PRG-RAM / battery-backed SRAM ($6000-$7FFF)
299            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize],
300
301            // Cartridge PRG-ROM (mapper-controlled)
302            0x8000..=0xFFFF => self.mapper.read_prg(addr),
303
304            // Unmapped regions ($4018-$401F, $4020-$5FFF)
305            _ => 0, // Open bus
306        }
307    }
308
309    fn write(&mut self, addr: u16, value: u8) {
310        match addr {
311            // 2KB internal RAM, mirrored 4 times
312            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize] = value,
313
314            // PPU registers, mirrored every 8 bytes
315            0x2000..=0x3FFF => {
316                let ppu_addr = 0x2000 + (addr & 0x0007);
317                // Pass closure to write CHR to mapper
318                let mapper = &mut *self.mapper;
319                self.ppu.write_register(ppu_addr, value, |chr_addr, val| {
320                    mapper.write_chr(chr_addr, val);
321                });
322            }
323
324            // APU registers
325            0x4000..=0x4013 | 0x4015 | 0x4017 => self.apu.write_register(addr, value),
326
327            // OAM DMA
328            0x4014 => self.start_oam_dma(value),
329
330            // Controller strobe (affects BOTH controllers)
331            0x4016 => {
332                self.controller1.write_strobe(value);
333                self.controller2.write_strobe(value);
334            }
335
336            // PRG-RAM / battery-backed SRAM ($6000-$7FFF)
337            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize] = value,
338
339            // Cartridge PRG-ROM / mapper registers ($8000-$FFFF)
340            0x8000..=0xFFFF => self.mapper.write_prg(addr, value),
341
342            // Unmapped regions ($4018-$401F, $4020-$5FFF)
343            _ => {}
344        }
345    }
346
347    fn peek(&self, addr: u16) -> u8 {
348        match addr {
349            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize],
350            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize],
351            0x8000..=0xFFFF => self.mapper.read_prg(addr),
352            _ => 0,
353        }
354    }
355}

362    fn create_test_bus() -> Bus {
363        // Create a minimal ROM for testing
364        let mut rom_data = vec![0; 16 + 16384]; // Header + 16KB PRG
365                                                // iNES header
366        rom_data[0..4].copy_from_slice(b"NES\x1A");
367        rom_data[4] = 1; // 1 PRG bank
368        rom_data[5] = 0; // 0 CHR banks (use CHR-RAM)
369        rom_data[6] = 0; // Horizontal mirroring, NROM
370
371        let rom = Rom::load(&rom_data).unwrap();
372        let mapper = Box::new(Nrom::new(&rom));
373
374        Bus::new(mapper)
375    }
376
377    #[test]
378    fn test_ram_read_write() {
379        let mut bus = create_test_bus();
380
381        bus.write(0x0000, 0x42);
382        assert_eq!(bus.read(0x0000), 0x42);
383
384        // Test mirroring
385        assert_eq!(bus.read(0x0800), 0x42);
386        assert_eq!(bus.read(0x1000), 0x42);
387        assert_eq!(bus.read(0x1800), 0x42);
388    }
389
390    #[test]
391    fn test_ppu_register_mirroring() {
392        let mut bus = create_test_bus();
393
394        // Write to PPUCTRL
395        bus.write(0x2000, 0x80);
396
397        // Should be mirrored every 8 bytes
398        bus.write(0x2008, 0x00);
399        bus.write(0x3000, 0x00);
400        bus.write(0x3FF8, 0x00);
401    }

403    #[test]
404    fn test_controller_strobe() {
405        let mut bus = create_test_bus();
406
407        // Set button on controller 1
408        bus.controller1.set_button(crate::input::Button::A, true);
409
410        // Strobe
411        bus.write(0x4016, 0x01);
412        bus.write(0x4016, 0x00);
413
414        // Read should return A button state
415        let value = bus.read(0x4016);
416        assert_eq!(value & 0x01, 1);
417    }

419    #[test]
420    #[allow(clippy::cast_possible_truncation)]
421    fn test_oam_dma() {
422        let mut bus = create_test_bus();
423
424        // Write test data to RAM
425        for i in 0..256_u16 {
426            bus.write(0x0200 + i, i as u8);
427        }
428
429        // Initiate DMA
430        bus.write(0x4014, 0x02);
431
432        assert!(bus.dma_active());
433
434        // Run DMA to completion
435        let mut cycles = 0;
436        while !bus.tick_dma() {
437            cycles += 1;
438            assert!(cycles <= 600, "DMA didn't complete");
439        }
440
441        assert!(!bus.dma_active());
442        assert!((512..=514).contains(&cycles));
443    }
