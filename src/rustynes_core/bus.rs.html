1//! NES memory bus implementation.
2//!
3//! This module implements the NES bus which routes CPU memory access to various
4//! components: internal RAM, PPU registers, APU registers, controllers, and cartridge.
5//!
6//! # Memory Map
7//!
8//! ```text
9//! $0000-$07FF: 2KB internal RAM
10//! $0800-$1FFF: Mirrors of $0000-$07FF (3Ã—)
11//! $2000-$2007: PPU registers
12//! $2008-$3FFF: Mirrors of $2000-$2007
13//! $4000-$4013: APU and I/O registers
14//! $4014:       OAM DMA register
15//! $4015:       APU status register
16//! $4016:       Controller 1 data / strobe
17//! $4017:       Controller 2 data / APU frame counter
18//! $4018-$401F: APU/I/O test mode registers (normally disabled)
19//! $4020-$FFFF: Cartridge space (mapper-controlled)
20//! ```
21

28/// NES memory bus connecting CPU to all components
29#[allow(clippy::struct_excessive_bools)]
30pub struct Bus {
31    /// 2KB internal RAM
32    ram: [u8; 0x800],
33
34    /// 8KB PRG-RAM / battery-backed SRAM ($6000-$7FFF)
35    /// Used by many test ROMs to report results
36    prg_ram: [u8; 0x2000],
37
38    /// Picture Processing Unit
39    pub ppu: Ppu,
40
41    /// Audio Processing Unit
42    pub apu: Apu,
43
44    /// Cartridge mapper
45    pub mapper: Box<dyn Mapper>,
46
47    /// Controller 1
48    pub controller1: Controller,
49
50    /// Controller 2
51    pub controller2: Controller,
52
53    /// OAM DMA state
54    dma_page: u8,
55    dma_addr: u8,
56    dma_data: u8,
57    dma_dummy_cycles: u8, // Number of dummy cycles remaining (1 or 2 depending on alignment)
58    dma_transfer: bool,
59    dma_write: bool,
60
61    /// CPU cycle counter for odd/even tracking (for DMA timing)
62    cpu_cycles: u64,
63
64    /// NMI pending from PPU (set during `on_cpu_cycle`, cleared by Console)
65    nmi_pending: bool,
66
67    /// Frame complete flag (set during `on_cpu_cycle`, cleared by Console)
68    frame_complete: bool,
69
70    /// DMC DMA stall cycles pending (set during `on_cpu_cycle`, cleared by Console)
71    /// When DMC performs DMA, it steals CPU cycles (typically 3-4 cycles)
72    dmc_stall_cycles: u8,
73
74    /// Last value on data bus (for open bus behavior)
75    /// Unmapped addresses return this value; controllers mix it with their data
76    last_bus_value: u8,

94    /// Create a new bus with the given mapper and custom audio sample rate
95    ///
96    /// # Arguments
97    ///
98    /// * `mapper` - Cartridge mapper implementation
99    /// * `sample_rate` - Audio output sample rate (e.g., 44100 or 48000 Hz)
100    ///
101    /// # Returns
102    ///
103    /// New Bus instance with all components initialized
104    #[must_use]
105    pub fn with_sample_rate(mapper: Box<dyn Mapper>, sample_rate: u32) -> Self {
106        let mirroring = mapper.mirroring();
107
108        // Convert mapper Mirroring to PPU Mirroring
109        let ppu_mirroring = match mirroring {
110            rustynes_mappers::Mirroring::Horizontal => PpuMirroring::Horizontal,
111            rustynes_mappers::Mirroring::Vertical => PpuMirroring::Vertical,
112            rustynes_mappers::Mirroring::SingleScreenLower => PpuMirroring::SingleScreenLower,
113            rustynes_mappers::Mirroring::SingleScreenUpper => PpuMirroring::SingleScreenUpper,
114            rustynes_mappers::Mirroring::FourScreen => PpuMirroring::FourScreen,
115        };
116
117        Self {
118            ram: [0; 0x800],
119            prg_ram: [0; 0x2000], // Initialize PRG-RAM to 0 (matches internal RAM)
120            ppu: Ppu::new(ppu_mirroring),
121            apu: Apu::with_sample_rate(sample_rate),
122            mapper,
123            controller1: Controller::new(),
124            controller2: Controller::new(),
125            dma_page: 0,
126            dma_addr: 0,
127            dma_data: 0,
128            dma_dummy_cycles: 0,
129            dma_transfer: false,
130            dma_write: false,
131            cpu_cycles: 0,
132            nmi_pending: false,
133            frame_complete: false,
134            dmc_stall_cycles: 0,
135            last_bus_value: 0,
136        }
137    }

139    /// Check if DMA transfer is active
140    ///
141    /// # Returns
142    ///
143    /// true if OAM DMA is in progress, false otherwise
144    #[must_use]
145    pub fn dma_active(&self) -> bool {
146        self.dma_transfer
147    }
148
149    /// Execute one cycle of DMA transfer
150    ///
151    /// OAM DMA takes 513 or 514 CPU cycles:
152    /// - 513 cycles: Started on even CPU cycle (1 dummy + 512 transfer cycles)
153    /// - 514 cycles: Started on odd CPU cycle (2 dummy cycles for alignment + 512 transfer cycles)
154    /// - Transfer: 256 alternating read/write cycles (512 total)
155    ///
156    /// # Returns
157    ///
158    /// true if DMA is complete, false if still in progress
159    pub fn tick_dma(&mut self) -> bool {
160        if !self.dma_transfer {
161            return true;
162        }
163
164        // Dummy cycle(s) for alignment
165        // If started on odd cycle, we need 2 dummy cycles
166        // If started on even cycle, we need 1 dummy cycle
167        if self.dma_dummy_cycles > 0 {
168            self.dma_dummy_cycles -= 1;
169            self.cpu_cycles += 1;
170            return false;
171        }
172
173        if self.dma_write {
174            // Write cycle
175            self.ppu.write_register(0x2004, self.dma_data, |_, _| {});
176            self.dma_write = false;
177            self.cpu_cycles += 1;
178
179            // Advance to next byte
180            self.dma_addr = self.dma_addr.wrapping_add(1);
181
182            // Check if transfer is complete (256 bytes transferred)
183            if self.dma_addr == 0 {
184                self.dma_transfer = false;
185                true
186            } else {
187                false
188            }
189        } else {
190            // Read cycle
191            let addr = u16::from(self.dma_page) << 8 | u16::from(self.dma_addr);
192            self.dma_data = self.read_for_dma(addr);
193            self.dma_write = true;
194            self.cpu_cycles += 1;
195            false
196        }
197    }
198
199    /// Read memory for DMA (doesn't trigger side effects)
200    ///
201    /// This is separate from regular `read()` to avoid triggering PPU/APU side effects
202    /// during DMA operations.
203    fn read_for_dma(&self, addr: u16) -> u8 {
204        match addr {
205            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize],
206            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize],
207            0x4020..=0xFFFF => self.mapper.read_prg(addr),
208            _ => 0, // DMA from PPU/APU registers returns open bus
209        }
210    }

212    /// Initiate OAM DMA transfer
213    ///
214    /// # Arguments
215    ///
216    /// * `page` - High byte of source address (e.g., $02 for $0200-$02FF)
217    ///
218    /// # Timing
219    ///
220    /// OAM DMA takes 513 or 514 CPU cycles:
221    /// - 513 cycles if started on an even CPU cycle (1 dummy + 512 transfer)
222    /// - 514 cycles if started on an odd CPU cycle (2 dummy + 512 transfer)
223    ///
224    /// The extra cycle on odd alignment ensures reads happen on even cycles.
225    fn start_oam_dma(&mut self, page: u8) {
226        self.dma_page = page;
227        self.dma_addr = 0;
228        self.dma_transfer = true;
229        self.dma_write = false;
230
231        // Determine number of dummy cycles based on CPU cycle parity
232        // Odd cycle: need 2 dummy cycles for alignment (514 total)
233        // Even cycle: need 1 dummy cycle (513 total)
234        self.dma_dummy_cycles = if (self.cpu_cycles % 2) == 1 { 2 } else { 1 };
235    }

237    /// Increment CPU cycle counter (called from console after each CPU instruction)
238    ///
239    /// This tracks odd/even cycles for DMA timing precision.
240    ///
241    /// # Arguments
242    ///
243    /// * `cycles` - Number of CPU cycles to add
244    pub fn add_cpu_cycles(&mut self, cycles: u8) {
245        self.cpu_cycles += u64::from(cycles);
246    }

248    /// Reset the bus and all components
249    pub fn reset(&mut self) {
250        self.ram = [0; 0x800];
251        self.prg_ram.fill(0); // Initialize PRG-RAM to 0
252        self.ppu.reset();
253        self.apu.reset();
254        self.controller1.reset();
255        self.controller2.reset();
256        self.dma_transfer = false;
257        self.cpu_cycles = 0;
258        self.nmi_pending = false;
259        self.frame_complete = false;
260        self.dmc_stall_cycles = 0;
261        self.last_bus_value = 0;
262    }

293    /// Step PPU by one dot with CHR ROM access from mapper
294    ///
295    /// This method bridges the PPU and mapper, allowing the PPU to fetch
296    /// pattern table data (CHR ROM) during background and sprite rendering.
297    ///
298    /// # Returns
299    ///
300    /// Tuple of (`frame_complete`, `nmi`):
301    /// - `frame_complete`: true if a complete frame was just rendered
302    /// - `nmi`: true if NMI should be triggered (`VBlank` start with NMI enabled)
303    #[inline]
304    pub fn step_ppu(&mut self) -> (bool, bool) {
305        // Borrow mapper immutably for CHR reads while PPU is borrowed mutably
306        // This works because they are separate fields of the struct
307        let mapper = &*self.mapper;
308        self.ppu.step_with_chr(|addr| mapper.read_chr(addr))
309    }

311    /// Take and clear the pending NMI flag.
312    ///
313    /// This is set by `on_cpu_cycle()` when PPU asserts NMI.
314    /// Console should call this after `cpu.tick()` to check if NMI needs handling.
315    ///
316    /// # Returns
317    ///
318    /// true if NMI was pending, false otherwise
319    #[inline]
320    pub fn take_nmi(&mut self) -> bool {
321        let pending = self.nmi_pending;
322        self.nmi_pending = false;
323        pending
324    }
325
326    /// Take and clear the frame complete flag.
327    ///
328    /// This is set by `on_cpu_cycle()` when PPU completes a frame.
329    /// Console should call this to detect when a new frame is ready.
330    ///
331    /// # Returns
332    ///
333    /// true if a frame was completed, false otherwise
334    #[inline]
335    pub fn take_frame_complete(&mut self) -> bool {
336        let complete = self.frame_complete;
337        self.frame_complete = false;
338        complete
339    }
340
341    /// Take and clear pending DMC DMA stall cycles.
342    ///
343    /// This is set by `on_cpu_cycle()` when DMC performs DMA.
344    /// Console should call this after `cpu.tick()` to check if CPU needs to stall.
345    ///
346    /// During DMC DMA stalls:
347    /// - CPU is halted (no instruction execution)
348    /// - PPU continues running (3 dots per stall cycle)
349    /// - APU continues running (1 cycle per stall cycle)
350    /// - Mapper is clocked (1 cycle per stall cycle)
351    ///
352    /// # Returns
353    ///
354    /// Number of stall cycles (0-4, typically 3 when DMA occurs)
355    #[inline]
356    pub fn take_dmc_stall_cycles(&mut self) -> u8 {
357        let stalls = self.dmc_stall_cycles;
358        self.dmc_stall_cycles = 0;
359        stalls
360    }

363impl CpuBus for Bus {
364    fn read(&mut self, addr: u16) -> u8 {
365        let value = match addr {
366            // 2KB internal RAM, mirrored 4 times
367            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize],
368
369            // PPU registers, mirrored every 8 bytes
370            0x2000..=0x3FFF => {
371                let ppu_addr = 0x2000 + (addr & 0x0007);
372                // Pass closure to read CHR from mapper
373                let mapper = &*self.mapper;
374                self.ppu
375                    .read_register(ppu_addr, |chr_addr| mapper.read_chr(chr_addr))
376            }
377
378            // APU and I/O registers
379            0x4000..=0x4015 => self.apu.read_register(addr),
380
381            // Controller 1 - bits 0-4 from controller, bits 5-7 from open bus
382            0x4016 => self.controller1.read() | (self.last_bus_value & 0xE0),
383
384            // Controller 2 (note: $4017 write goes to APU, read goes to controller)
385            // bits 0-4 from controller, bits 5-7 from open bus
386            0x4017 => self.controller2.read() | (self.last_bus_value & 0xE0),
387
388            // Unmapped APU/IO test registers ($4018-$401F) and expansion ROM area ($4020-$5FFF)
389            // Both return open bus value
390            0x4018..=0x5FFF => self.last_bus_value,
391
392            // PRG-RAM / battery-backed SRAM ($6000-$7FFF)
393            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize],
394
395            // Cartridge PRG-ROM (mapper-controlled)
396            0x8000..=0xFFFF => self.mapper.read_prg(addr),
397        };
398
399        // Track last bus value for open bus behavior
400        self.last_bus_value = value;
401        value
402    }
403
404    fn write(&mut self, addr: u16, value: u8) {
405        // Track last bus value for open bus behavior
406        // Writes also put a value on the data bus
407        self.last_bus_value = value;
408
409        match addr {
410            // 2KB internal RAM, mirrored 4 times
411            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize] = value,
412
413            // PPU registers, mirrored every 8 bytes
414            0x2000..=0x3FFF => {
415                let ppu_addr = 0x2000 + (addr & 0x0007);
416                // Pass closure to write CHR to mapper
417                let mapper = &mut *self.mapper;
418                self.ppu.write_register(ppu_addr, value, |chr_addr, val| {
419                    mapper.write_chr(chr_addr, val);
420                });
421            }
422
423            // APU registers
424            0x4000..=0x4013 | 0x4015 | 0x4017 => self.apu.write_register(addr, value),
425
426            // OAM DMA
427            0x4014 => self.start_oam_dma(value),
428
429            // Controller strobe (affects BOTH controllers)
430            0x4016 => {
431                self.controller1.write_strobe(value);
432                self.controller2.write_strobe(value);
433            }
434
435            // PRG-RAM / battery-backed SRAM ($6000-$7FFF)
436            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize] = value,
437
438            // Cartridge PRG-ROM / mapper registers ($8000-$FFFF)
439            0x8000..=0xFFFF => self.mapper.write_prg(addr, value),
440
441            // Unmapped regions ($4018-$401F, $4020-$5FFF)
442            _ => {}
443        }
444    }
445
446    /// Step PPU, APU, and mapper before each CPU memory access.
447    ///
448    /// This is the core of cycle-accurate emulation. For NTSC:
449    /// - PPU runs at 3x CPU clock (3 PPU dots per CPU cycle)
450    /// - APU runs at CPU clock
451    /// - Mapper is clocked once per CPU cycle (for IRQ timing)
452    ///
453    /// Called BEFORE each CPU read/write to ensure PPU, APU, and mapper are in
454    /// the correct state when the CPU observes memory. This is critical
455    /// for accurate `VBlank` flag ($2002) timing and mapper IRQ precision.
456    ///
457    /// NMI, `frame_complete`, and DMC stall signals are captured and can be
458    /// retrieved via `take_nmi()`, `take_frame_complete()`, and `take_dmc_stall_cycles()`.
459    #[inline]
460    fn on_cpu_cycle(&mut self) {
461        // Step PPU 3 times (3 PPU dots per CPU cycle for NTSC)
462        for _ in 0..3 {
463            let (frame_complete, nmi) = self.step_ppu();
464            if nmi {
465                self.nmi_pending = true;
466            }
467            if frame_complete {
468                self.frame_complete = true;
469            }
470        }
471
472        // Step APU once (1 APU cycle per CPU cycle)
473        // APU internally divides this further for its frame sequencer
474        // APU returns DMC DMA stall cycles if DMC performed a DMA fetch
475        let dmc_stalls = self.apu.step();
476        if dmc_stalls > 0 {
477            // Accumulate stall cycles (can happen multiple times if CPU stalls span instructions)
478            self.dmc_stall_cycles = self.dmc_stall_cycles.saturating_add(dmc_stalls);
479        }
480
481        // Clock mapper once per CPU cycle (for cycle-based IRQ timing)
482        // This is critical for VRC mappers and other cycle-counting mappers
483        self.mapper.clock(1);
484
485        // Track CPU cycles for DMA alignment
486        self.cpu_cycles += 1;
487    }
488
489    fn peek(&self, addr: u16) -> u8 {
490        match addr {
491            0x0000..=0x1FFF => self.ram[(addr & 0x07FF) as usize],
492            0x6000..=0x7FFF => self.prg_ram[(addr - 0x6000) as usize],
493            0x8000..=0xFFFF => self.mapper.read_prg(addr),
494            _ => 0,
495        }
496    }
497}

504    fn create_test_bus() -> Bus {
505        // Create a minimal ROM for testing
506        let mut rom_data = vec![0; 16 + 16384]; // Header + 16KB PRG
507        // iNES header
508        rom_data[0..4].copy_from_slice(b"NES\x1A");
509        rom_data[4] = 1; // 1 PRG bank
510        rom_data[5] = 0; // 0 CHR banks (use CHR-RAM)
511        rom_data[6] = 0; // Horizontal mirroring, NROM
512
513        let rom = Rom::load(&rom_data).unwrap();
514        let mapper = Box::new(Nrom::new(&rom));
515
516        Bus::new(mapper)
517    }

625    fn test_controller_open_bus_bits() {
626        let mut bus = create_test_bus();
627
628        // Strobe controllers first
629        bus.write(0x4016, 0x01);
630        bus.write(0x4016, 0x00);

633        // Note: writes update open bus too, so we must read to set it after strobe
634        bus.write(0x0000, 0xE0);
635        let _ = bus.read(0x0000); // Open bus now 0xE0
636
637        // Controller read should mix open bus bits 5-7 with controller data bits 0-4
638        // With no buttons pressed, controller returns 0x00 for bit 0
639        // Result bits 5-7 should come from open bus (0xE0)
640        let value = bus.read(0x4016);
641        assert_eq!(value & 0xE0, 0xE0); // Upper 3 bits from open bus
642
643        // Set open bus again before reading controller 2
644        // (the previous controller read updated open bus to the read value)
645        let _ = bus.read(0x0000); // Open bus now 0xE0 again
646        let value2 = bus.read(0x4017);
647        assert_eq!(value2 & 0xE0, 0xE0);
648    }
