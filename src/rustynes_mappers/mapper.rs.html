1//! Mapper trait definition.
2//!
3//! The Mapper trait defines the interface for all NES cartridge mapper implementations.
4//! Mappers handle bank switching, mirroring control, and special features like IRQ generation.
5
6use crate::Mirroring;
7
8/// NES cartridge mapper interface.
9///
10/// This trait defines the contract for all mapper implementations. Mappers are responsible for:
11///
12/// - **PRG-ROM Banking**: Mapping CPU address space ($8000-$FFFF) to PRG-ROM banks
13/// - **CHR Banking**: Mapping PPU address space ($0000-$1FFF) to CHR-ROM/RAM banks
14/// - **Mirroring Control**: Determining nametable mirroring mode
15/// - **IRQ Generation**: Triggering interrupts for scanline effects (MMC3, MMC5, etc.)
16/// - **Battery-Backed RAM**: Providing save game storage (SRAM)
17///
18/// # Safety
19///
20/// All mapper implementations must be `Send` to allow safe transfer between threads.
21/// Mappers should avoid internal mutability beyond what's necessary for register writes.
22///
23/// # Examples
24///
25/// ```no_run
26/// use rustynes_mappers::{Mapper, Mirroring};
27///
28/// // Simple passthrough mapper (NROM)
29/// struct Nrom {
30///     prg_rom: Vec<u8>,
31///     chr_rom: Vec<u8>,
32///     mirroring: Mirroring,
33/// }
34///
35/// impl Mapper for Nrom {
36///     fn read_prg(&self, addr: u16) -> u8 {
37///         let offset = (addr - 0x8000) as usize % self.prg_rom.len();
38///         self.prg_rom[offset]
39///     }
40///
41///     fn write_prg(&mut self, _addr: u16, _value: u8) {
42///         // NROM has no writable registers
43///     }
44///
45///     fn read_chr(&self, addr: u16) -> u8 {
46///         self.chr_rom[addr as usize]
47///     }
48///
49///     fn write_chr(&mut self, _addr: u16, _value: u8) {
50///         // NROM has CHR-ROM (not writable)
51///     }
52///
53///     fn mirroring(&self) -> Mirroring {
54///         self.mirroring
55///     }
56///
57///     fn mapper_number(&self) -> u16 {
58///         0
59///     }
60///
61///     fn clone_mapper(&self) -> Box<dyn Mapper> {
62///         Box::new(Nrom {
63///             prg_rom: self.prg_rom.clone(),
64///             chr_rom: self.chr_rom.clone(),
65///             mirroring: self.mirroring,
66///         })
67///     }
68/// }
69/// ```
70pub trait Mapper: Send {
71    /// Read from PRG-ROM address space ($8000-$FFFF).
72    ///
73    /// # Arguments
74    ///
75    /// * `addr` - CPU address in range $8000-$FFFF
76    ///
77    /// # Returns
78    ///
79    /// Byte value at the given address after bank mapping
80    ///
81    /// # Panics
82    ///
83    /// Implementations may panic if `addr < 0x8000` (invalid PRG address).
84    fn read_prg(&self, addr: u16) -> u8;
85
86    /// Write to PRG address space ($8000-$FFFF).
87    ///
88    /// This is primarily used for writing to mapper registers. Writes may:
89    /// - Change PRG/CHR banking
90    /// - Update mirroring mode
91    /// - Configure IRQ settings
92    /// - Write to PRG-RAM (if present and not write-protected)
93    ///
94    /// # Arguments
95    ///
96    /// * `addr` - CPU address in range $8000-$FFFF
97    /// * `value` - Byte value to write
98    ///
99    /// # Panics
100    ///
101    /// Implementations may panic if `addr < 0x8000` (invalid PRG address).
102    fn write_prg(&mut self, addr: u16, value: u8);
103
104    /// Read from CHR address space ($0000-$1FFF).
105    ///
106    /// # Arguments
107    ///
108    /// * `addr` - PPU address in range $0000-$1FFF
109    ///
110    /// # Returns
111    ///
112    /// Byte value at the given address after bank mapping
113    ///
114    /// # Panics
115    ///
116    /// Implementations may panic if `addr > 0x1FFF` (invalid CHR address).
117    fn read_chr(&self, addr: u16) -> u8;
118
119    /// Write to CHR address space ($0000-$1FFF).
120    ///
121    /// Only functional for CHR-RAM. CHR-ROM cartridges typically ignore writes.
122    ///
123    /// # Arguments
124    ///
125    /// * `addr` - PPU address in range $0000-$1FFF
126    /// * `value` - Byte value to write
127    ///
128    /// # Panics
129    ///
130    /// Implementations may panic if `addr > 0x1FFF` (invalid CHR address).
131    fn write_chr(&mut self, addr: u16, value: u8);
132
133    /// Get current nametable mirroring mode.
134    ///
135    /// # Returns
136    ///
137    /// Current mirroring mode. May change dynamically for mappers with mirroring control.
138    fn mirroring(&self) -> Mirroring;
139
140    /// Check if an IRQ is pending.
141    ///
142    /// Used by mappers with IRQ support (MMC3, MMC5, VRC, etc.) to signal the CPU.
143    ///
144    /// # Returns
145    ///
146    /// `true` if the mapper is asserting IRQ, `false` otherwise.
147    ///
148    /// # Default Implementation
149    ///
150    /// Returns `false` for mappers without IRQ support.
151    fn irq_pending(&self) -> bool {
152        false
153    }
154
155    /// Clear the IRQ flag.
156    ///
157    /// Called when the CPU acknowledges the IRQ (typically after reading $FFFC/FFFD).
158    ///
159    /// # Default Implementation
160    ///
161    /// No-op for mappers without IRQ support.
162    fn clear_irq(&mut self) {}
163
164    /// Clock the mapper for a number of CPU cycles.
165    ///
166    /// Used by mappers with cycle-based timers (MMC3 alternate IRQ mode, FDS, etc.).
167    ///
168    /// # Arguments
169    ///
170    /// * `cycles` - Number of CPU cycles elapsed
171    ///
172    /// # Default Implementation
173    ///
174    /// No-op for mappers without cycle-based features.
175    fn clock(&mut self, _cycles: u8) {}
176
177    /// Notify mapper of PPU A12 rising edge.
178    ///
179    /// Used by MMC3 and compatible mappers for scanline IRQ counting.
180    /// Called when PPU address bit 12 transitions from 0 to 1.
181    ///
182    /// # Default Implementation
183    ///
184    /// No-op for mappers without A12-based IRQ.
185    fn ppu_a12_edge(&mut self) {}
186
187    /// Get reference to battery-backed SRAM.
188    ///
189    /// # Returns
190    ///
191    /// `Some(&[u8])` if the mapper has battery-backed SRAM, `None` otherwise.
192    ///
193    /// # Default Implementation
194    ///
195    /// Returns `None` for mappers without SRAM.
196    fn sram(&self) -> Option<&[u8]> {
197        None
198    }
199
200    /// Get mutable reference to battery-backed SRAM.
201    ///
202    /// # Returns
203    ///
204    /// `Some(&mut [u8])` if the mapper has battery-backed SRAM, `None` otherwise.
205    ///
206    /// # Default Implementation
207    ///
208    /// Returns `None` for mappers without SRAM.
209    fn sram_mut(&mut self) -> Option<&mut [u8]> {
210        None
211    }
212
213    /// Get the mapper number (iNES/NES 2.0 format).
214    ///
215    /// # Returns
216    ///
217    /// iNES mapper number (0-4095 for NES 2.0, 0-255 for iNES 1.0)
218    fn mapper_number(&self) -> u16;
219
220    /// Get the submapper number (NES 2.0 format only).
221    ///
222    /// # Returns
223    ///
224    /// Submapper number (0-15), or 0 if not applicable.
225    ///
226    /// # Default Implementation
227    ///
228    /// Returns 0 (no submapper).
229    fn submapper(&self) -> u8 {
230        0
231    }
232
233    /// Clone the mapper state.
234    ///
235    /// This is used for save states and debugging. Implementations should return
236    /// a boxed clone of the current mapper state.
237    ///
238    /// # Returns
239    ///
240    /// Boxed clone of the mapper
241    fn clone_mapper(&self) -> Box<dyn Mapper>;
242}

244#[cfg(test)]
245mod tests {
246    use super::*;
247
248    // Simple test mapper to verify trait implementation
249    struct TestMapper {
250        prg_rom: Vec<u8>,
251        chr_ram: Vec<u8>,
252        mirroring: Mirroring,
253    }
254
255    impl Mapper for TestMapper {
256        fn read_prg(&self, addr: u16) -> u8 {
257            assert!(addr >= 0x8000, "Invalid PRG address");
258            let offset = (addr - 0x8000) as usize;
259            self.prg_rom[offset % self.prg_rom.len()]
260        }
261
262        fn write_prg(&mut self, addr: u16, _value: u8) {
263            assert!(addr >= 0x8000, "Invalid PRG address");
264        }
265
266        fn read_chr(&self, addr: u16) -> u8 {
267            assert!(addr <= 0x1FFF, "Invalid CHR address");
268            self.chr_ram[addr as usize]
269        }
270
271        fn write_chr(&mut self, addr: u16, value: u8) {
272            assert!(addr <= 0x1FFF, "Invalid CHR address");
273            self.chr_ram[addr as usize] = value;
274        }
275
276        fn mirroring(&self) -> Mirroring {
277            self.mirroring
278        }
279
280        fn mapper_number(&self) -> u16 {
281            0
282        }
283
284        fn clone_mapper(&self) -> Box<dyn Mapper> {
285            Box::new(TestMapper {
286                prg_rom: self.prg_rom.clone(),
287                chr_ram: self.chr_ram.clone(),
288                mirroring: self.mirroring,
289            })
290        }
291    }
292
293    #[test]
294    fn test_mapper_trait_implementation() {
295        let mut mapper = TestMapper {
296            prg_rom: vec![0x42; 0x8000],
297            chr_ram: vec![0; 0x2000],
298            mirroring: Mirroring::Horizontal,
299        };
300
301        // Test PRG read
302        assert_eq!(mapper.read_prg(0x8000), 0x42);
303        assert_eq!(mapper.read_prg(0xFFFF), 0x42);
304
305        // Test PRG write (should not panic)
306        mapper.write_prg(0x8000, 0xFF);
307
308        // Test CHR read/write
309        mapper.write_chr(0x1000, 0x55);
310        assert_eq!(mapper.read_chr(0x1000), 0x55);
311
312        // Test mirroring
313        assert_eq!(mapper.mirroring(), Mirroring::Horizontal);
314
315        // Test default implementations
316        assert!(!mapper.irq_pending());
317        mapper.clear_irq();
318        mapper.clock(1);
319        mapper.ppu_a12_edge();
320        assert!(mapper.sram().is_none());
321        assert!(mapper.sram_mut().is_none());
322        assert_eq!(mapper.submapper(), 0);
323    }

328        assert_send::<Box<dyn Mapper>>();
329    }
