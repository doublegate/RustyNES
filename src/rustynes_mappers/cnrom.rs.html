1//! Mapper 3: CNROM
2//!
3//! CNROM is a simple discrete logic mapper featuring:
4//! - Fixed 16KB or 32KB PRG-ROM (no banking)
5//! - Switchable 8KB CHR-ROM banks
6//! - Single register write to any address $8000-$FFFF
7//!
8//! # Hardware Details
9//!
10//! - **PRG-ROM**: 16KB or 32KB (fixed, no banking)
11//! - **CHR-ROM**: Up to 256KB (32 banks of 8KB)
12//! - **Mirroring**: Fixed horizontal or vertical (hardware)
13//! - **Bus Conflicts**: Yes (write value must match ROM data)
14//!
15//! # Games
16//!
17//! - Arkanoid
18//! - Paperboy
19//! - Solomon's Key
20//! - Gradius
21//! - Cybernoid
22

25/// CNROM mapper implementation (Mapper 3).
26#[derive(Clone)]
27pub struct Cnrom {
28    /// PRG-ROM data (16KB or 32KB).
29    prg_rom: Vec<u8>,
30
31    /// CHR-ROM data.
32    chr_rom: Vec<u8>,
33
34    /// Nametable mirroring mode (fixed by hardware).
35    mirroring: Mirroring,
36
37    /// Currently selected CHR bank.
38    chr_bank: u8,
39
40    /// Number of 8KB CHR banks.
41    chr_banks: usize,
42}
43
44impl Cnrom {
45    /// Create a new CNROM mapper from a ROM.
46    ///
47    /// # Arguments
48    ///
49    /// * `rom` - Loaded ROM file
50    ///
51    /// # Panics
52    ///
53    /// Panics if:
54    /// - PRG-ROM size is not 16KB or 32KB
55    /// - CHR-ROM is empty (CNROM requires CHR-ROM, not CHR-RAM)
56    /// - CHR-ROM size is not a multiple of 8KB
57    #[must_use]
58    pub fn new(rom: &Rom) -> Self {
59        // Validate PRG-ROM size
60        assert!(
61            rom.prg_rom.len() == 16384 || rom.prg_rom.len() == 32768,
62            "CNROM requires 16KB or 32KB PRG-ROM, got {} bytes",
63            rom.prg_rom.len()
64        );
65
66        // CNROM requires CHR-ROM
67        assert!(
68            !rom.chr_rom.is_empty(),
69            "CNROM requires CHR-ROM (got CHR-RAM)"
70        );
71
72        // Validate CHR-ROM size
73        assert_eq!(
74            rom.chr_rom.len() % 8192,
75            0,
76            "CHR-ROM size must be a multiple of 8KB"
77        );
78
79        let chr_banks = rom.chr_rom.len() / 8192;
80
81        Self {
82            prg_rom: rom.prg_rom.clone(),
83            chr_rom: rom.chr_rom.clone(),
84            mirroring: rom.header.mirroring,
85            chr_bank: 0,
86            chr_banks,
87        }
88    }
89}

91impl Mapper for Cnrom {
92    fn read_prg(&self, addr: u16) -> u8 {
93        debug_assert!(addr >= 0x8000, "Invalid PRG address: ${addr:04X}");
94
95        let offset = (addr - 0x8000) as usize;
96
97        // Handle mirroring for 16KB PRG-ROM
98        let masked_offset = if self.prg_rom.len() == 16384 {
99            offset & 0x3FFF // Mirror 16KB to fill 32KB space
100        } else {
101            offset
102        };
103
104        self.prg_rom[masked_offset]
105    }
106
107    fn write_prg(&mut self, _addr: u16, value: u8) {
108        // CHR bank select register
109        // In real hardware, this would AND with ROM data (bus conflicts)
110        // Only the lower 2 bits are used for bank selection (max 4 banks = 32KB)
111        // But some games have more, so we mask to the actual number of banks
112        self.chr_bank = value;
113    }
114
115    fn read_chr(&self, addr: u16) -> u8 {
116        debug_assert!(addr <= 0x1FFF, "Invalid CHR address: ${addr:04X}");
117
118        let bank = (self.chr_bank as usize) % self.chr_banks;
119        let offset = addr as usize;
120        self.chr_rom[bank * 8192 + offset]
121    }
122
123    fn write_chr(&mut self, _addr: u16, _value: u8) {
124        // CHR-ROM is not writable
125    }
126
127    fn mirroring(&self) -> Mirroring {
128        self.mirroring
129    }
130
131    fn mapper_number(&self) -> u16 {
132        3
133    }
134
135    fn clone_mapper(&self) -> Box<dyn Mapper> {
136        Box::new(self.clone())
137    }
138}

146    fn create_test_rom(prg_size: usize, chr_banks: usize, mirroring: Mirroring) -> Rom {
147        let header = RomHeader {
148            prg_rom_size: prg_size,
149            chr_rom_size: chr_banks * 8192,
150            mapper_number: 3,
151            submapper: 0,
152            mirroring,
153            has_battery: false,
154            has_trainer: false,
155            nes2_format: false,
156            prg_ram_size: 0,
157            prg_nvram_size: 0,
158            chr_ram_size: 0,
159            chr_nvram_size: 0,
160        };
161
162        Rom {
163            header,
164            trainer: None,
165            prg_rom: vec![0; prg_size],
166            chr_rom: (0..chr_banks * 8192).map(|i| (i / 8192) as u8).collect(),
167        }
168    }

218    #[test]
219    fn test_chr_bank_switching() {
220        let rom = create_test_rom(32768, 4, Mirroring::Horizontal);
221        let mut mapper = Cnrom::new(&rom);
222
223        // Initial bank is 0
224        assert_eq!(mapper.chr_bank, 0);
225        assert_eq!(mapper.read_chr(0x0000), 0);
226
227        // Switch to bank 1
228        mapper.write_prg(0x8000, 1);
229        assert_eq!(mapper.chr_bank, 1);
230        assert_eq!(mapper.read_chr(0x0000), 1);
231
232        // Switch to bank 2
233        mapper.write_prg(0x8000, 2);
234        assert_eq!(mapper.chr_bank, 2);
235        assert_eq!(mapper.read_chr(0x0000), 2);
236
237        // Switch to bank 3
238        mapper.write_prg(0x8000, 3);
239        assert_eq!(mapper.chr_bank, 3);
240        assert_eq!(mapper.read_chr(0x0000), 3);
241    }
