1//! Welcome screen view.
2//!
3//! This is the initial screen shown when the application launches with no ROM loaded.
4//! Provides a clean interface for loading ROMs via file dialog.
5
6use iced::widget::{button, column, container, text, Column};
7use iced::{Element, Length};
8
9use crate::app::RustyNes;
10use crate::message::Message;
11use crate::theme::RustyPalette;
12
13/// Render the Welcome view
14pub fn view(_app: &RustyNes) -> Element<'_, Message> {
15    let palette = RustyPalette::dark();
16
17    // Title with "Power Red" color (primary)
18    let title = text("RustyNES").size(48).color(palette.primary);
19
20    // Subtitle
21    let subtitle = text("Next-Generation NES Emulator")
22        .size(20)
23        .color(palette.accent);
24
25    // Version info
26    let version = text(format!("v{}", env!("CARGO_PKG_VERSION"))).size(14);
27
28    // Open ROM button
29    let open_button = button(text("Open ROM").size(18))
30        .padding(15)
31        .on_press(Message::OpenFileDialog);
32
33    // Main content column
34    let content: Column<Message> = column![title, subtitle, version, open_button,]
35        .spacing(20)
36        .padding(40)
37        .align_x(iced::alignment::Horizontal::Center);
38
39    // Centered container
40    container(content)
41        .width(Length::Fill)
42        .height(Length::Fill)
43        .center_x(Length::Fill)
44        .center_y(Length::Fill)
45        .into()
46}
