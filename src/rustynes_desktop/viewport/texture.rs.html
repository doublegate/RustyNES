1//! NES framebuffer texture management.
2//!
3//! Handles the GPU texture for the NES framebuffer (256×240 resolution)
4//! with efficient RGB to RGBA conversion and texture upload.
5
6// Re-export wgpu types from iced to ensure version compatibility
7use iced::widget::shader::wgpu;
8use wgpu::{Device, Queue, Sampler, Texture, TextureFormat, TextureView};
9
10/// NES framebuffer texture (256×240 RGB888 → RGBA8 GPU)
11#[allow(dead_code)] // Will be used for custom wgpu rendering pipeline
12pub struct NesTexture {
13    texture: Texture,
14    view: TextureView,
15    width: u32,
16    height: u32,
17    /// Pre-allocated RGBA buffer (avoid allocations in hot path)
18    rgba_buffer: Vec<u8>,
19}
20
21#[allow(dead_code)] // Will be used for custom wgpu rendering pipeline
22impl NesTexture {
23    /// NES native resolution width
24    pub const NES_WIDTH: u32 = 256;
25    /// NES native resolution height
26    pub const NES_HEIGHT: u32 = 240;
27
28    /// Create a new NES texture
29    pub fn new(device: &Device) -> Self {
30        let texture = device.create_texture(&wgpu::TextureDescriptor {
31            label: Some("NES Framebuffer"),
32            size: wgpu::Extent3d {
33                width: Self::NES_WIDTH,
34                height: Self::NES_HEIGHT,
35                depth_or_array_layers: 1,
36            },
37            mip_level_count: 1,
38            sample_count: 1,
39            dimension: wgpu::TextureDimension::D2,
40            format: TextureFormat::Rgba8UnormSrgb,
41            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
42            view_formats: &[],
43        });
44
45        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
46
47        // Pre-allocate RGBA buffer to avoid allocations during texture updates
48        let rgba_buffer = vec![0u8; (Self::NES_WIDTH * Self::NES_HEIGHT * 4) as usize];
49
50        Self {
51            texture,
52            view,
53            width: Self::NES_WIDTH,
54            height: Self::NES_HEIGHT,
55            rgba_buffer,
56        }
57    }
58
59    /// Upload NES framebuffer (RGB888) to GPU texture
60    ///
61    /// # Performance
62    /// - Converts RGB to RGBA in-place using pre-allocated buffer
63    /// - Target: <1ms upload time
64    ///
65    /// # Panics
66    /// Panics if framebuffer size is not 256×240×3 bytes (184,320 bytes)
67    pub fn update(&mut self, queue: &Queue, framebuffer: &[u8]) {
68        assert_eq!(
69            framebuffer.len(),
70            (Self::NES_WIDTH * Self::NES_HEIGHT * 3) as usize,
71            "Invalid framebuffer size: expected {} bytes, got {}",
72            Self::NES_WIDTH * Self::NES_HEIGHT * 3,
73            framebuffer.len()
74        );
75
76        // Convert RGB888 → RGBA8888 in-place (no allocations)
77        for (i, chunk) in framebuffer.chunks_exact(3).enumerate() {
78            let offset = i * 4;
79            self.rgba_buffer[offset] = chunk[0]; // R
80            self.rgba_buffer[offset + 1] = chunk[1]; // G
81            self.rgba_buffer[offset + 2] = chunk[2]; // B
82            self.rgba_buffer[offset + 3] = 255; // A (opaque)
83        }
84
85        queue.write_texture(
86            wgpu::ImageCopyTexture {
87                texture: &self.texture,
88                mip_level: 0,
89                origin: wgpu::Origin3d::ZERO,
90                aspect: wgpu::TextureAspect::All,
91            },
92            &self.rgba_buffer,
93            wgpu::ImageDataLayout {
94                offset: 0,
95                bytes_per_row: Some(Self::NES_WIDTH * 4),
96                rows_per_image: Some(Self::NES_HEIGHT),
97            },
98            wgpu::Extent3d {
99                width: Self::NES_WIDTH,
100                height: Self::NES_HEIGHT,
101                depth_or_array_layers: 1,
102            },
103        );
104    }
105
106    /// Get texture view for binding
107    pub fn view(&self) -> &TextureView {
108        &self.view
109    }
110
111    /// Get texture dimensions
112    pub fn dimensions(&self) -> (u32, u32) {
113        (self.width, self.height)
114    }
115}

117/// Nearest-neighbor sampler for pixel-perfect NES rendering
118#[allow(dead_code)] // Will be used for custom wgpu rendering pipeline
119pub struct NesSampler {
120    sampler: Sampler,
121}
122
123#[allow(dead_code)] // Will be used for custom wgpu rendering pipeline
124impl NesSampler {
125    /// Create a new nearest-neighbor sampler
126    pub fn new(device: &Device) -> Self {
127        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
128            label: Some("NES Sampler (Nearest)"),
129            address_mode_u: wgpu::AddressMode::ClampToEdge,
130            address_mode_v: wgpu::AddressMode::ClampToEdge,
131            address_mode_w: wgpu::AddressMode::ClampToEdge,
132            mag_filter: wgpu::FilterMode::Nearest, // Pixel-perfect scaling
133            min_filter: wgpu::FilterMode::Nearest,
134            mipmap_filter: wgpu::FilterMode::Nearest,
135            ..Default::default()
136        });
137
138        Self { sampler }
139    }
140
141    /// Get sampler for binding
142    pub fn sampler(&self) -> &Sampler {
143        &self.sampler
144    }
145}
