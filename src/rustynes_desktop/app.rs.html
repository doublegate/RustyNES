52// === NES Timing Constants (NTSC) ===
53// Master clock: 21.477272 MHz
54// CPU: 1.789773 MHz (master / 12)
55// PPU: 5.369318 MHz (master / 4) = 3x CPU clock
56
57/// CPU frequency in Hz (NTSC).
58const CPU_FREQUENCY_NTSC: f64 = 1_789_773.0;
59
60/// Average CPU cycles per frame (odd frames: 29781, even frames: 29780).
61const CPU_CYCLES_PER_FRAME: f64 = 29780.5;
62
63/// CPU cycles for even frames.
64#[allow(dead_code)]
65const CPU_CYCLES_EVEN_FRAME: u32 = 29780;
66
67/// CPU cycles for odd frames.
68#[allow(dead_code)]
69const CPU_CYCLES_ODD_FRAME: u32 = 29781;
70
71/// Target frame rate (NTSC): `CPU_FREQUENCY_NTSC` / `CPU_CYCLES_PER_FRAME` = ~60.0988 Hz.
72const TARGET_FPS: f64 = CPU_FREQUENCY_NTSC / CPU_CYCLES_PER_FRAME;
73
74/// Frame duration in nanoseconds (~16,639,265 ns).
75const FRAME_DURATION: Duration = Duration::from_nanos((1_000_000_000.0 / TARGET_FPS) as u64);

187    /// Number of frames to pre-warm the emulator after ROM load.
188    /// The PPU needs ~40 frames before rendering visible content.
189    /// Pre-warming eliminates the grey screen on startup.
190    const PREWARM_FRAMES: u32 = 50;
191
192    /// Load a ROM file into a new console instance with pre-warming.
193    ///
194    /// Pre-warming runs the emulator for `PREWARM_FRAMES` before returning,
195    /// eliminating the grey screen that occurs during PPU initialization.
196    /// Audio samples generated during pre-warming are cleared.
197    fn load_rom_with_prewarm(path: &PathBuf, sample_rate: u32) -> anyhow::Result<Console> {
198        let rom_data = std::fs::read(path)?;
199        let mut console = Console::from_rom_bytes_with_sample_rate(&rom_data, sample_rate)
200            .map_err(|e| anyhow::anyhow!("{e}"))?;
201
202        // Pre-warm the emulator to eliminate grey screen on startup.
203        // The first ~40 frames have garbage PPU data during initialization.
204        info!(
205            "Pre-warming emulator for {} frames to eliminate grey screen...",
206            Self::PREWARM_FRAMES
207        );
208        for _ in 0..Self::PREWARM_FRAMES {
209            console.step_frame();
210        }
211
212        // Clear audio samples generated during pre-warm to avoid audio burst
213        console.clear_audio_samples();
214
215        Ok(console)
216    }

218    /// Update the NES texture from the console framebuffer.
219    ///
220    /// The PPU outputs palette indices (0-63). This method converts them
221    /// to RGBA using the NES system palette for display.
222    fn update_texture(&mut self, ctx: &egui::Context) {
223        // Get pixel data from console or use placeholder
224        if let Some(console) = &self.console {
225            let palette_indices = console.framebuffer();
226
227            // Convert palette indices to RGBA
228            // PPU outputs 256x240 palette indices (0-63), we convert to RGBA
229            for (i, &palette_idx) in palette_indices.iter().enumerate() {
230                // Clamp palette index to valid range (0-63)
231                let idx = (palette_idx & 0x3F) as usize;
232                let (r, g, b) = NES_PALETTE[idx];
233
234                // Write RGBA to framebuffer (4 bytes per pixel)
235                let offset = i * 4;
236                if offset + 3 < self.framebuffer.len() {
237                    self.framebuffer[offset] = r;
238                    self.framebuffer[offset + 1] = g;
239                    self.framebuffer[offset + 2] = b;
240                    self.framebuffer[offset + 3] = 255; // Full alpha
241                }
242            }
243        } else {
244            // Dark blue placeholder when no ROM is loaded
245            for pixel in self.framebuffer.chunks_exact_mut(4) {
246                pixel[0] = 32;
247                pixel[1] = 32;
248                pixel[2] = 64;
249                pixel[3] = 255;
250            }
251        }
252
253        // Create or update the texture
254        let image = ColorImage::from_rgba_unmultiplied([NES_WIDTH, NES_HEIGHT], &self.framebuffer);
255
256        if let Some(texture) = &mut self.nes_texture {
257            texture.set(image, TextureOptions::NEAREST);
258        } else {
259            self.nes_texture =
260                Some(ctx.load_texture("nes_framebuffer", image, TextureOptions::NEAREST));
261        }
262    }

264    /// Run emulation for one frame.
265    ///
266    /// Returns a speed adjustment factor for A/V synchronization.
267    /// - 0.99: Buffer low, slow down slightly
268    /// - 1.00: Buffer healthy, no adjustment
269    /// - 1.01: Buffer high, speed up slightly
270    fn run_frame(&mut self) -> f32 {
271        let mut speed_adjustment = 1.0;
272
273        if let Some(console) = &mut self.console {
274            // Update controller input BEFORE emulation (minimizes input latency)
275            console.set_controller_1(self.input.player1_buttons());
276            console.set_controller_2(self.input.player2_buttons());
277
278            // Run one frame using cycle-accurate mode
279            console.step_frame_accurate();
280
281            // Get audio samples and queue them with adaptive sync
282            if let Some(audio) = &mut self.audio {
283                let samples = console.audio_samples();
284                if !samples.is_empty() {
285                    // Use adaptive sync to maintain audio/video synchronization
286                    let (_, adjustment) = audio.queue_samples_with_sync(samples);
287                    speed_adjustment = adjustment;
288                }
289                console.clear_audio_samples();
290            }
291        }
292
293        speed_adjustment
294    }

296    /// Handle keyboard input for special keys and shortcuts.
297    fn handle_special_keys(&mut self, ctx: &egui::Context) {
298        ctx.input(|i| {
299            let ctrl = i.modifiers.ctrl;
300
301            // Ctrl+O: Open ROM
302            if ctrl && i.key_pressed(egui::Key::O) {
303                self.open_file_dialog();
304            }
305
306            // Ctrl+P: Toggle Pause
307            if ctrl && i.key_pressed(egui::Key::P) && self.console.is_some() {
308                self.paused = !self.paused;
309                self.gui_state
310                    .set_status(StatusMessage::info(if self.paused {
311                        "Emulation paused"
312                    } else {
313                        "Emulation resumed"
314                    }));
315                info!("Console {}", if self.paused { "paused" } else { "resumed" });
316            }
317
318            // Ctrl+R: Reset
319            if ctrl
320                && i.key_pressed(egui::Key::R)
321                && let Some(console) = &mut self.console
322            {
323                console.reset();
324                self.gui_state
325                    .set_status(StatusMessage::info("Console reset"));
326                info!("Console reset");
327            }
328
329            // Ctrl+Q: Quit
330            if ctrl && i.key_pressed(egui::Key::Q) {
331                std::process::exit(0);
332            }
333
334            // Ctrl+,: Settings (comma key)
335            if ctrl && i.key_pressed(egui::Key::Comma) {
336                self.gui_state.settings_open = true;
337            }
338
339            // F3: Toggle pause (legacy)
340            if i.key_pressed(egui::Key::F3) && self.console.is_some() {
341                self.paused = !self.paused;
342                self.gui_state
343                    .set_status(StatusMessage::info(if self.paused {
344                        "Emulation paused"
345                    } else {
346                        "Emulation resumed"
347                    }));
348                info!("Console {}", if self.paused { "paused" } else { "resumed" });
349            }
350
351            // F2: Reset (legacy)
352            if i.key_pressed(egui::Key::F2)
353                && let Some(console) = &mut self.console
354            {
355                console.reset();
356                self.gui_state
357                    .set_status(StatusMessage::info("Console reset"));
358                info!("Console reset");
359            }
360
361            // F1: Toggle debug mode
362            if i.key_pressed(egui::Key::F1) {
363                self.config.debug.enabled = !self.config.debug.enabled;
364                self.gui_state
365                    .set_status(StatusMessage::info(if self.config.debug.enabled {
366                        "Debug mode enabled"
367                    } else {
368                        "Debug mode disabled"
369                    }));
370            }
371
372            // Escape: Toggle menu or close dialogs
373            if i.key_pressed(egui::Key::Escape) {
374                if self.gui_state.show_welcome {
375                    self.gui_state.show_welcome = false;
376                    self.config.first_run = false;
377                } else if self.gui_state.error_message.is_some() {
378                    self.gui_state.error_message = None;
379                } else if self.gui_state.confirm_action.is_some() {
380                    self.gui_state.confirm_action = None;
381                } else if self.gui_state.settings_open {
382                    self.gui_state.settings_open = false;
383                } else if self.gui_state.show_shortcuts {
384                    self.gui_state.show_shortcuts = false;
385                } else {
386                    self.gui_state.toggle_menu();
387                }
388            }
389
390            // M: Toggle mute
391            if i.key_pressed(egui::Key::M)
392                && !i.modifiers.ctrl
393                && let Some(audio) = &self.audio
394            {
395                audio.toggle_mute();
396                let muted = audio.is_muted();
397                self.gui_state.set_status(StatusMessage::info(if muted {
398                    "Audio muted"
399                } else {
400                    "Audio unmuted"
401                }));
402            }
403        });
404    }

444    /// Handle keyboard input for NES controller using configured bindings.
445    ///
446    /// Uses direct key state polling to ensure game controller input always works,
447    /// even when egui widgets have keyboard focus (e.g., modal dialogs).
448    fn handle_controller_keys(&mut self, ctx: &egui::Context) {
449        use crate::input::{KeyCode, egui_key_to_keycode};
450
451        ctx.input(|i| {
452            // Poll all controller-relevant keys directly using key_down()
453            // This ensures input works regardless of egui focus state
454            let controller_keys = [
455                // D-pad
456                egui::Key::ArrowUp,
457                egui::Key::ArrowDown,
458                egui::Key::ArrowLeft,
459                egui::Key::ArrowRight,
460                // Action buttons (common bindings)
461                egui::Key::X, // A
462                egui::Key::Z, // B
463                // Start/Select
464                egui::Key::Enter, // Start
465                // WASD alternative for player 2
466                egui::Key::W,
467                egui::Key::A,
468                egui::Key::S,
469                egui::Key::D,
470                // Number keys for player 2 buttons
471                egui::Key::Num1,
472                egui::Key::Num2,
473            ];
474
475            for &key in &controller_keys {
476                let pressed = i.key_down(key);
477                if let Some(keycode) = egui_key_to_keycode(key) {
478                    self.input.handle_key_1(keycode, pressed);
479                    self.input.handle_key_2(keycode, pressed);
480                }
481            }
482
483            // Handle shift keys for Select button (not exposed as regular Key events)
484            // Check modifiers for shift state changes
485            let shift_pressed = i.modifiers.shift;
486
487            // Track shift state for player 1 (ShiftRight is default Select)
488            self.input.handle_key_1(KeyCode::ShiftRight, shift_pressed);
489            self.input.handle_key_1(KeyCode::ShiftLeft, shift_pressed);
490
491            // Track shift state for player 2
492            self.input.handle_key_2(KeyCode::ShiftRight, shift_pressed);
493            self.input.handle_key_2(KeyCode::ShiftLeft, shift_pressed);
494        });
495    }

497    /// Handle file drops.
498    fn handle_dropped_files(&mut self, ctx: &egui::Context) {
499        // Get sample rate before the closure to avoid borrowing issues
500        let sample_rate = self.audio_sample_rate();
501
502        ctx.input(|i| {
503            for file in &i.raw.dropped_files {
504                if let Some(path) = &file.path {
505                    let rom_name = path
506                        .file_name()
507                        .and_then(|n| n.to_str())
508                        .unwrap_or("Unknown")
509                        .to_string();
510
511                    info!("File dropped: {}", path.display());
512                    match Self::load_rom_with_prewarm(path, sample_rate) {
513                        Ok(console) => {
514                            self.console = Some(console);
515                            self.paused = false;
516                            self.config.recent_roms.add(path.clone());
517                            self.gui_state.rom_name = Some(rom_name.clone());
518                            self.gui_state
519                                .set_status(StatusMessage::success(format!("Loaded: {rom_name}")));
520                        }
521                        Err(e) => {
522                            self.gui_state.set_error(format!("Failed to load ROM: {e}"));
523                            error!("Failed to load dropped file: {e}");
524                        }
525                    }
526                }
527            }
528        });
529    }

532impl eframe::App for NesApp {
533    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
534        // Request continuous repaint for smooth emulation
535        ctx.request_repaint();
536
537        // Apply theme changes if config changed
538        if self.config.video.theme != self.last_theme {
539            gui::apply_theme(ctx, &self.config);
540            self.last_theme = self.config.video.theme;
541        }
542
543        // Handle input
544        self.handle_special_keys(ctx);
545        self.handle_controller_keys(ctx);
546        self.handle_dropped_files(ctx);
547
548        // Poll gamepads
549        self.input.poll_gamepads();
550
551        // Frame timing and emulation with A/V sync
552        let now = Instant::now();
553        let delta = now - self.last_frame;
554        self.last_frame = now;
555
556        if !self.paused {
557            self.accumulator += delta;
558
559            // Apply speed adjustment for A/V sync (0.99x - 1.01x)
560            // This slightly adjusts the effective frame duration to maintain audio sync
561            #[allow(
562                clippy::cast_sign_loss,
563                clippy::cast_possible_truncation,
564                clippy::cast_precision_loss
565            )]
566            let adjusted_duration = Duration::from_nanos(
567                (FRAME_DURATION.as_nanos() as f64 / f64::from(self.speed_adjustment)) as u64,
568            );
569
570            // Run emulation to catch up
571            while self.accumulator >= adjusted_duration {
572                self.accumulator -= adjusted_duration;
573                self.speed_adjustment = self.run_frame();
574            }
575        }
576
577        // Update the framebuffer texture
578        self.update_texture(ctx);
579
580        // Render GUI (menu bar and overlays)
581        gui::render(
582            ctx,
583            &mut self.gui_state,
584            &mut self.config,
585            &mut self.console,
586            &self.audio,
587            &mut self.paused,
588        );
589
590        // Render the NES display in the central panel
591        egui::CentralPanel::default().show(ctx, |ui| {
592            // Calculate the best fit size while maintaining aspect ratio
593            let available_size = ui.available_size();
594
595            // NES aspect ratio is 256:240 (1.067), but with 8:7 pixel aspect correction it's ~1.14
596            let nes_aspect = if self.config.video.pixel_aspect_correction {
597                256.0 * (8.0 / 7.0) / 240.0
598            } else {
599                256.0 / 240.0
600            };
601
602            let (display_width, display_height) = {
603                let width_from_height = available_size.y * nes_aspect;
604                let height_from_width = available_size.x / nes_aspect;
605
606                if width_from_height <= available_size.x {
607                    (width_from_height, available_size.y)
608                } else {
609                    (available_size.x, height_from_width)
610                }
611            };
612
613            // Center the display using vertical and horizontal centering
614            ui.vertical_centered(|ui| {
615                ui.add_space((available_size.y - display_height) / 2.0);
616                if let Some(texture) = &self.nes_texture {
617                    ui.image(egui::load::SizedTexture::new(
618                        texture.id(),
619                        egui::vec2(display_width, display_height),
620                    ));
621                }
622            });
623        });
624    }
625
626    fn on_exit(&mut self, _gl: Option<&eframe::glow::Context>) {
627        // Save configuration on exit
628        if let Err(e) = self.config.save() {
629            error!("Failed to save config: {e}");
630        }
631        info!("RustyNES exiting");
632    }
