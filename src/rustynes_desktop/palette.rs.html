76/// Convert NES palette index to RGB color
77///
78/// # Arguments
79///
80/// * `index` - Palette index (0-63)
81///
82/// # Returns
83///
84/// RGB color as [R, G, B]
85#[must_use]
86pub fn palette_to_rgb(index: u8) -> [u8; 3] {
87    NES_PALETTE[usize::from(index & 0x3F)]
88}
89
90/// Convert NES framebuffer (palette indices) to RGB buffer
91///
92/// # Arguments
93///
94/// * `framebuffer` - Palette indices (256×240 = 61,440 bytes)
95///
96/// # Returns
97///
98/// RGB buffer (256×240×3 = 184,320 bytes)
99#[must_use]
100pub fn framebuffer_to_rgb(framebuffer: &[u8]) -> Vec<u8> {
101    let mut rgb = Vec::with_capacity(256 * 240 * 3);
102
103    for &index in framebuffer {
104        let color = palette_to_rgb(index);
105        rgb.push(color[0]); // R
106        rgb.push(color[1]); // G
107        rgb.push(color[2]); // B
108    }
109
110    rgb
111}

113#[cfg(test)]
114mod tests {
115    use super::*;
116
117    #[test]
118    fn test_palette_to_rgb() {
119        // Test first color (gray)
120        assert_eq!(palette_to_rgb(0x00), [0x66, 0x66, 0x66]);
121
122        // Test white
123        assert_eq!(palette_to_rgb(0x20), [0xFF, 0xFE, 0xFF]);
124
125        // Test palette wrapping (indices > 63 wrap)
126        assert_eq!(palette_to_rgb(0x40), palette_to_rgb(0x00));
127        assert_eq!(palette_to_rgb(0xFF), palette_to_rgb(0x3F));
128    }
129
130    #[test]
131    fn test_framebuffer_to_rgb() {
132        // Test small framebuffer
133        let framebuffer = vec![0x00, 0x20, 0x0D];
134        let rgb = framebuffer_to_rgb(&framebuffer);
135
136        assert_eq!(rgb.len(), 9); // 3 pixels × 3 bytes
137        assert_eq!(&rgb[0..3], &[0x66, 0x66, 0x66]); // First pixel
138        assert_eq!(&rgb[3..6], &[0xFF, 0xFE, 0xFF]); // Second pixel
139        assert_eq!(&rgb[6..9], &[0x00, 0x00, 0x00]); // Third pixel
140    }
