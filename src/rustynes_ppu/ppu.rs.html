1//! Main PPU (Picture Processing Unit) implementation
2//!
3//! The Ricoh 2C02 PPU is responsible for generating the video output
4//! for the NES. It renders 256×240 pixel frames at 60Hz (NTSC).
5//!
6//! # Memory Map (PPU address space)
7//!
8//! ```text
9//! $0000-$0FFF: Pattern Table 0 (CHR ROM/RAM, via mapper)
10//! $1000-$1FFF: Pattern Table 1 (CHR ROM/RAM, via mapper)
11//! $2000-$2FFF: Nametables (internal VRAM with mirroring)
12//! $3F00-$3F1F: Palette RAM
13//! ```
14//!
15//! # CPU Registers ($2000-$2007)
16//!
17//! ```text
18//! $2000: PPUCTRL   - Control register
19//! $2001: PPUMASK   - Mask register
20//! $2002: PPUSTATUS - Status register
21//! $2003: OAMADDR   - OAM address
22//! $2004: OAMDATA   - OAM data
23//! $2005: PPUSCROLL - Scroll position
24//! $2006: PPUADDR   - VRAM address
25//! $2007: PPUDATA   - VRAM data
26//! ```
27

119    /// Read from PPU register (CPU memory map $2000-$2007)
120    ///
121    /// # Arguments
122    ///
123    /// * `addr` - Register address
124    /// * `read_chr` - Callback to read CHR memory (mapper) for addresses < $2000
125    pub fn read_register<F: FnMut(u16) -> u8>(&mut self, addr: u16, mut read_chr: F) -> u8 {
126        match addr & 0x07 {
127            // $2000: PPUCTRL (write-only) -> return open bus (do not refresh)
128            0 => self.open_bus_latch,
129
130            // $2001: PPUMASK (write-only) -> return open bus (do not refresh)
131            1 => self.open_bus_latch,
132
133            // $2002: PPUSTATUS
134            2 => {
135                // Reading $2002 only drives bits 7-5. Bits 4-0 are open bus (undriven).
136                // Therefore, we do NOT refresh the decay counter, so bits 4-0 continue to decay.
137                // (Technically bits 7-5 are refreshed, but our model has one counter).
138
139                let status = self.status.bits();
140
141                // Race condition: Reading $2002 on the exact cycle VBlank is set
142                // suppresses the NMI. This happens at scanline 241, dot 1.
143                if self.timing.scanline() == 241 && self.timing.dot() == 1 {
144                    self.nmi_pending = false;
145                }
146
147                self.status.clear_vblank(); // Reading clears VBlank flag
148                self.scroll.read_ppustatus(); // Reset write latch
149
150                // Return status (bits 7-5) + open bus (bits 4-0)
151                let result = (status & 0xE0) | (self.open_bus_latch & 0x1F);
152
153                // Update latch with result (actually only bits 7-5 are new, 4-0 are preserved)
154                self.open_bus_latch = result;
155
156                result
157            }
158
159            // $2003: OAMADDR (write-only) -> return open bus (do not refresh)
160            3 => self.open_bus_latch,
161
162            // $2004: OAMDATA
163            4 => {
164                // Reading refreshes open bus
165                self.refresh_open_bus();
166
167                let data = self.oam.read();
168                // Reading OAMDATA does NOT reliably update open bus on all revisions,
169                // but usually it does. Most emulators update it.
170                self.open_bus_latch = data;
171                data
172            }
173
174            // $2005: PPUSCROLL (write-only) -> return open bus (do not refresh)
175            5 => self.open_bus_latch,
176
177            // $2006: PPUADDR (write-only) -> return open bus (do not refresh)
178            6 => self.open_bus_latch,
179
180            // $2007: PPUDATA
181            7 => {
182                // Reading refreshes open bus
183                self.refresh_open_bus();
184
185                let addr = self.scroll.vram_addr();
186
187                // Read from CHR (mapper) or VRAM/Palette
188                let data = if (addr & 0x3FFF) < 0x2000 {
189                    read_chr(addr & 0x3FFF)
190                } else {
191                    self.vram.read(addr)
192                };
193
194                // Buffered read behavior
195                let result = if addr >= 0x3F00 {
196                    // Palette reads are immediate
197                    // Bits 7-6 are open bus (from decay latch)
198                    let pal_data = (data & 0x3F) | (self.open_bus_latch & 0xC0);
199
200                    // Reading the palette also updates the VRAM read buffer with
201                    // the contents of the mirrored nametable address ($2F00-$2FFF)
202                    self.vram_read_buffer = self.vram.read(addr - 0x1000);
203
204                    pal_data
205                } else {
206                    // Normal reads return previous buffer
207                    let buffered = self.vram_read_buffer;
208                    self.vram_read_buffer = data;
209                    buffered
210                };
211
212                // Increment VRAM address
213                let increment = self.ctrl.vram_increment();
214                self.scroll.increment_vram(increment);
215
216                // Update open bus latch with the value put on the bus
217                self.open_bus_latch = result;
218
219                result
220            }
221            _ => unreachable!(),
222        }
223    }
224    /// Write to PPU register (CPU memory map $2000-$2007)
225    ///
226    /// # Arguments
227    ///
228    /// * `addr` - Register address
229    /// * `value` - Value to write
230    /// * `write_chr` - Callback to write CHR memory (mapper) for addresses < $2000
231    pub fn write_register<F: FnMut(u16, u8)>(&mut self, addr: u16, value: u8, mut write_chr: F) {
232        // Writing to any register updates the open bus latch and refreshes decay
233        self.open_bus_latch = value;
234        self.refresh_open_bus();
235
236        match addr & 0x07 {
237            // $2000: PPUCTRL
238            0 => {
239                self.ctrl = PpuCtrl::from_bits_truncate(value);
240                self.scroll.write_ppuctrl(value);
241
242                // Check NMI enable
243                if self.ctrl.nmi_enabled() && self.status.in_vblank() {
244                    self.nmi_pending = true;
245                }
246            }
247
248            // $2001: PPUMASK
249            1 => {
250                self.mask = PpuMask::from_bits_truncate(value);
251            }
252
253            // $2002: PPUSTATUS (read-only)
254            2 => {}
255
256            // $2003: OAMADDR
257            3 => {
258                self.oam.set_addr(value);
259            }
260
261            // $2004: OAMDATA
262            4 => {
263                self.oam.write(value);
264            }
265
266            // $2005: PPUSCROLL
267            5 => {
268                // Detect mid-scanline write for split-screen effects
269                if self.is_visible_rendering_position() {
270                    self.scroll.record_mid_scanline_write();
271                }
272                self.scroll.write_ppuscroll(value);
273            }
274
275            // $2006: PPUADDR
276            6 => {
277                // Detect mid-scanline write for split-screen effects
278                // The second write to $2006 copies t to v, which affects rendering
279                if self.is_visible_rendering_position() {
280                    self.scroll.record_mid_scanline_write();
281                }
282                self.scroll.write_ppuaddr(value);
283            }
284
285            // $2007: PPUDATA
286            7 => {
287                let addr = self.scroll.vram_addr();
288
289                // Write to CHR (mapper) or VRAM/Palette
290                if (addr & 0x3FFF) < 0x2000 {
291                    write_chr(addr & 0x3FFF, value);
292                } else {
293                    self.vram.write(addr, value);
294                }
295
296                // Increment VRAM address
297                let increment = self.ctrl.vram_increment();
298                self.scroll.increment_vram(increment);
299            }
300
301            _ => unreachable!(),
302        }
303    }

319    /// Step PPU by one dot with CHR ROM access
320    ///
321    /// This method allows the PPU to read pattern table data from the mapper's CHR ROM.
322    ///
323    /// # Arguments
324    ///
325    /// * `read_chr` - Function to read CHR ROM at a given address (0x0000-0x1FFF)
326    ///
327    /// # Returns
328    ///
329    /// (frame_complete, nmi_triggered)
330    #[inline]
331    #[allow(clippy::too_many_lines)] // PPU step naturally handles many timing states
332    pub fn step_with_chr<F: Fn(u16) -> u8>(&mut self, read_chr: F) -> (bool, bool) {
333        // Open bus decay
334        if self.decay_counter > 0 {
335            self.decay_counter -= 1;
336            if self.decay_counter == 0 {
337                self.open_bus_latch = 0;
338            }
339        }
340
341        let rendering_enabled = self.mask.rendering_enabled();
342
343        // Tick timing FIRST to advance to the next position
344        let frame_complete = self.timing.tick(rendering_enabled);
345
346        // Process delayed PPUADDR updates (hardware delays t->v copy by ~2 PPU cycles)
347        self.scroll.delayed_update();
348
349        let scanline = self.timing.scanline();
350        let dot = self.timing.dot();
351
352        // VBlank flag management (check AFTER tick)
353        if self.timing.is_vblank_set_dot() {
354            self.status.set_vblank();
355            if self.ctrl.nmi_enabled() {
356                self.nmi_pending = true;
357            }
358        }
359
360        if self.timing.is_vblank_clear_dot() {
361            self.status.clear_vblank();
362            self.status.clear_sprite_flags();
363            self.nmi_pending = false;
364            // Reset frame-specific scroll tracking for mid-scanline detection
365            self.scroll.start_frame();
366        }
367
368        // Rendering logic (visible and pre-render scanlines)
369        if rendering_enabled && self.timing.is_rendering_scanline() {
370            // Render pixel FIRST using current shift register state (visible scanlines only)
371            // This must happen BEFORE shifting registers for correct pixel alignment
372            if self.timing.is_visible_scanline() && self.timing.is_visible_dot() {
373                let x = dot - 1;
374                let y = scanline;
375                self.render_pixel(x as usize, y as usize);
376            }
377
378            // Background rendering - shift AFTER rendering
379            if self.timing.is_visible_dot() || self.timing.is_prefetch_dot() {
380                self.background.shift_registers();
381
382                // 8-dot tile fetch cycle
383                // Dots are 1-indexed: 1-256 visible, 321-336 prefetch
384                let fetch_dot = dot;
385                match fetch_dot % 8 {
386                    1 => {
387                        // Fetch nametable byte (tile index)
388                        let nt_addr = 0x2000 | (self.scroll.vram_addr() & 0x0FFF);
389                        let tile_index = self.vram.read(nt_addr);
390                        self.background.set_nametable_byte(tile_index);
391                    }
392                    3 => {
393                        // Fetch attribute byte
394                        let v = self.scroll.vram_addr();
395                        let attr_addr =
396                            0x23C0 | (v & 0x0C00) | ((v >> 4) & 0x38) | ((v >> 2) & 0x07);
397                        let attr_byte = self.vram.read(attr_addr);
398                        self.background.set_attribute_byte(
399                            attr_byte,
400                            self.scroll.coarse_x(),
401                            self.scroll.coarse_y(),
402                        );
403                    }
404                    5 => {
405                        // Fetch pattern table low byte
406                        let bg_base = self.ctrl.bg_table_addr();
407                        let tile_index = self.background.nametable_byte();
408                        let fine_y = self.scroll.fine_y();
409                        let pattern_addr = bg_base + u16::from(tile_index) * 16 + u16::from(fine_y);
410                        let pattern_low = read_chr(pattern_addr);
411                        self.background.set_pattern_low(pattern_low);
412                    }
413                    7 => {
414                        // Fetch pattern table high byte
415                        let bg_base = self.ctrl.bg_table_addr();
416                        let tile_index = self.background.nametable_byte();
417                        let fine_y = self.scroll.fine_y();
418                        let pattern_addr =
419                            bg_base + u16::from(tile_index) * 16 + u16::from(fine_y) + 8;
420                        let pattern_high = read_chr(pattern_addr);
421                        self.background.set_pattern_high(pattern_high);
422                    }
423                    0 => {
424                        // Load shift registers and increment coarse X
425                        self.background.load_shift_registers();
426                        self.scroll.increment_x();
427                    }
428                    _ => {}
429                }
430
431                // Increment Y at dot 256
432                if dot == 256 {
433                    self.scroll.increment_y();
434                }
435            }
436
437            // Sprite rendering
438            if self.timing.is_visible_dot() {
439                self.sprite_renderer.tick();
440            }
441
442            // Scrolling updates
443            if self.timing.is_hori_copy_dot() {
444                self.scroll.copy_horizontal();
445            }
446
447            if self.timing.is_vert_copy_range() {
448                self.scroll.copy_vertical();
449            }
450
451            // Sprite evaluation (visible and pre-render scanlines)
452            // Pre-render scanline (261) evaluates sprites for scanline 0
453            if self.timing.is_visible_scanline() || self.timing.is_prerender_scanline() {
454                if self.timing.is_sprite_eval_start() {
455                    self.sprite_evaluator.start_evaluation();
456                    self.secondary_oam.clear();
457                }
458
459                if self.timing.is_sprite_eval_range() {
460                    // Calculate target scanline for evaluation
461                    // For pre-render (261), target is scanline 0
462                    // For visible scanlines (0-239), target is scanline+1
463                    let target_scanline = if self.timing.is_prerender_scanline() {
464                        0
465                    } else {
466                        scanline + 1
467                    };
468
469                    self.sprite_evaluator.evaluate_step(
470                        self.oam.data(),
471                        target_scanline,
472                        self.ctrl.sprite_height(),
473                        &mut self.secondary_oam,
474                    );
475                }
476            }
477
478            // Sprite fetching (all rendering scanlines)
479            if self.timing.is_sprite_fetch_start() {
480                // Load sprites from secondary OAM into sprite renderer
481                let sprite_zero_in_range = self.sprite_evaluator.sprite_zero_in_range();
482                self.sprite_renderer
483                    .load_sprites(&self.secondary_oam, sprite_zero_in_range);
484            }
485
486            if self.timing.is_sprite_fetch_range() {
487                // Fetch sprite pattern data during dots 257-320 (8 dots per sprite, 8 sprites)
488                let fetch_cycle = dot - 257; // 0-63
489                let sprite_index = fetch_cycle / 8; // 0-7 (which sprite)
490                let fetch_step = fetch_cycle % 8; // 0-7 (which step in the 8-dot cycle)
491
492                // On step 7, fetch both pattern bytes and load into sprite renderer
493                // (simplified from hardware timing which fetches in steps 4 and 6)
494                if fetch_step == 7
495                    && let Some(sprite) = self.secondary_oam.get_sprite(sprite_index as u8)
496                {
497                    let sprite_height = self.ctrl.sprite_height();
498                    let tile_index = sprite.tile_index;
499
500                    // Calculate which row of the sprite to fetch
501                    // We're fetching for the next scanline (scanline+1 for visible, 0 for pre-render)
502                    // Sprite Y in OAM is "top of sprite minus 1", so for target scanline T:
503                    //   displayed_row = T - (sprite.y + 1) = (T - 1) - sprite.y
504                    // For visible scanlines: displayed_row = scanline - sprite.y
505                    // For pre-render (261): displayed_row = (0 - 1) - sprite.y (wrapping handled)
506                    let target_scanline = if self.timing.is_prerender_scanline() {
507                        0u16
508                    } else {
509                        scanline
510                    };
511                    let sprite_y = target_scanline.saturating_sub(sprite.y as u16);
512
513                    // Clamp sprite_y to valid range based on sprite height
514                    let max_row = (sprite_height - 1) as u16;
515                    let sprite_y = sprite_y.min(max_row);
516
517                    // Handle vertical flip
518                    let mut row = if sprite.attributes.flip_vertical() {
519                        max_row - sprite_y
520                    } else {
521                        sprite_y
522                    };
523
524                    // Calculate tile address based on sprite mode
525                    let pattern_addr_low = if sprite_height == 16 {
526                        // 8x16 sprites: bit 0 of tile index selects pattern table
527                        let sprite_table = (u16::from(tile_index) & 0x01) * 0x1000;
528                        // For rows 8-15, we need to access the next tile (add 16 bytes)
529                        if row >= 8 {
530                            row += 8; // Skip to next tile's data
531                        }
532                        // Use bits 7-1 of tile index as the tile pair base
533                        sprite_table | ((u16::from(tile_index) & 0xFE) << 4) | row
534                    } else {
535                        // 8x8 sprites: use PPUCTRL to select pattern table
536                        let sprite_base = self.ctrl.sprite_table_addr();
537                        sprite_base + u16::from(tile_index) * 16 + row
538                    };
539
540                    let mut pattern_low = read_chr(pattern_addr_low);
541
542                    // Fetch pattern table high byte
543                    let pattern_addr_high = pattern_addr_low + 8;
544                    let mut pattern_high = read_chr(pattern_addr_high);
545
546                    // Handle horizontal flip
547                    if sprite.attributes.flip_horizontal() {
548                        pattern_low = pattern_low.reverse_bits();
549                        pattern_high = pattern_high.reverse_bits();
550                    }
551
552                    // Load pattern data into sprite renderer
553                    self.sprite_renderer.load_sprite_pattern(
554                        sprite_index as u8,
555                        pattern_low,
556                        pattern_high,
557                    );
558                }
559            }
560        }
561
562        let nmi = self.nmi_pending;
563        if nmi {
564            self.nmi_pending = false;
565        }
566
567        (frame_complete, nmi)
568    }

570    /// Render a single pixel
571    #[inline]
572    fn render_pixel(&mut self, x: usize, y: usize) {
573        let mut bg_pixel = 0;
574        let mut bg_palette = 0;
575
576        // Left 8 pixel clipping flags
577        let clip_bg_left = x < 8 && !self.mask.show_bg_left();
578        let clip_sprites_left = x < 8 && !self.mask.show_sprites_left();
579
580        // Get background pixel (respecting left clipping)
581        if self.mask.show_background() && !clip_bg_left {
582            let fine_x = self.scroll.fine_x();
583            let (pixel, palette) = self.background.get_pixel(fine_x);
584            bg_pixel = pixel;
585            bg_palette = palette;
586        }
587
588        let mut sprite_pixel = 0;
589        let mut sprite_palette = 0;
590        let mut sprite_priority = false;
591        let mut sprite_zero = false;
592
593        // Get sprite pixel (respecting left clipping)
594        if self.mask.show_sprites()
595            && !clip_sprites_left
596            && let Some((pixel, palette, priority, is_sprite_zero)) =
597                self.sprite_renderer.get_pixel()
598        {
599            sprite_pixel = pixel;
600            sprite_palette = palette;
601            sprite_priority = priority;
602            sprite_zero = is_sprite_zero;
603        }
604
605        // Sprite 0 hit detection
606        // Note: Sprite 0 hit does NOT occur at x=255 (hardware quirk)
607        // and requires both background and sprite to be non-transparent
608        // Left clipping affects hit detection - if either is clipped, no hit
609        if sprite_zero && bg_pixel != 0 && sprite_pixel != 0 && x != 255 {
610            self.status.set_sprite_zero_hit();
611        }
612
613        // Multiplexing (determine final pixel)
614        // Returns (pixel, palette, is_sprite) to correctly select sprite vs background palette
615        let (final_pixel, final_palette, is_sprite) = if bg_pixel == 0 && sprite_pixel == 0 {
616            // Both transparent - use backdrop color (background palette 0)
617            (0, 0, false)
618        } else if bg_pixel == 0 {
619            // Background transparent - show sprite
620            (sprite_pixel, sprite_palette, true)
621        } else if sprite_pixel == 0 {
622            // Sprite transparent - show background
623            (bg_pixel, bg_palette, false)
624        } else {
625            // Both opaque - check priority
626            if sprite_priority {
627                (bg_pixel, bg_palette, false)
628            } else {
629                (sprite_pixel, sprite_palette, true)
630            }
631        };
632
633        // Read palette and write to frame buffer
634        // Background palettes: $3F00-$3F0F (addresses 0-15)
635        // Sprite palettes: $3F10-$3F1F (addresses 16-31)
636        let palette_base: u8 = if is_sprite { 16 } else { 0 };
637        let palette_addr = palette_base + (final_palette << 2) + final_pixel;
638        let color_index = self.vram.read_palette(palette_addr);
639
640        let offset = y * FRAME_WIDTH + x;
641        self.frame_buffer[offset] = color_index;
642    }

655    /// Reset to power-up state
656    pub fn reset(&mut self) {
657        self.ctrl = PpuCtrl::empty();
658        self.mask = PpuMask::empty();
659        self.status = PpuStatus::empty();
660        self.scroll = ScrollRegisters::new();

662        self.oam.reset();
663        self.background.reset();
664        self.sprite_renderer.reset();
665        self.timing.reset();
666        self.frame_buffer.fill(0);
667        self.vram_read_buffer = 0;
668        self.nmi_pending = false;
669    }

773        // Step PPU to apply delayed PPUADDR update (2 PPU cycles)
774        ppu.step();
775        ppu.step();

785        ppu.step();
786
787        // First read is buffered (returns garbage)
788        let _ = ppu.read_register(0x2007, |_| 0);
