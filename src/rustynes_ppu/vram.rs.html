1//! PPU VRAM (Video RAM) and palette RAM implementation
2//!
3//! The PPU has access to:
4//! - 2KB internal VRAM for nametables (mirrored to 4KB address space)
5//! - 32 bytes of palette RAM
6//! - External CHR ROM/RAM (accessed via mapper)
7//!
8//! # Address Space ($0000-$3FFF)
9//!
10//! ```text
11//! $0000-$0FFF: Pattern Table 0 (CHR ROM/RAM)
12//! $1000-$1FFF: Pattern Table 1 (CHR ROM/RAM)
13//! $2000-$23FF: Nametable 0
14//! $2400-$27FF: Nametable 1
15//! $2800-$2BFF: Nametable 2
16//! $2C00-$2FFF: Nametable 3
17//! $3000-$3EFF: Mirror of $2000-$2EFF
18//! $3F00-$3F1F: Palette RAM (32 bytes)
19//! $3F20-$3FFF: Mirror of $3F00-$3F1F
20//! ```
21//!
22//! # Nametable Mirroring
23//!
24//! The NES only has 2KB internal VRAM, so nametables are mirrored:
25//! - **Horizontal**: A A B B (vertical scrolling games)
26//! - **Vertical**: A B A B (horizontal scrolling games)
27//! - **Single-Screen**: A A A A or B B B B
28//! - **Four-Screen**: A B C D (mapper provides extra RAM)
29
30/// Nametable mirroring mode
31#[derive(Debug, Clone, Copy, PartialEq, Eq)]
32pub enum Mirroring {
33    /// Horizontal mirroring (vertical arrangement)
34    Horizontal,
35    /// Vertical mirroring (horizontal arrangement)
36    Vertical,
37    /// Single-screen mirroring (lower bank)
38    SingleScreenLower,
39    /// Single-screen mirroring (upper bank)
40    SingleScreenUpper,
41    /// Four-screen mirroring (mapper-provided RAM)
42    FourScreen,
43}
44
45/// PPU VRAM manager
46///
47/// Handles internal 2KB nametable RAM and 32-byte palette RAM.
48/// Pattern table access is delegated to the mapper.
49pub struct Vram {
50    /// Internal nametable RAM (2KB)
51    nametables: Vec<u8>,
52    /// Palette RAM (32 bytes)
53    palette: Vec<u8>,
54    /// Current mirroring mode
55    mirroring: Mirroring,
56    /// Four-screen mode extra RAM (optional, 4KB)
57    four_screen_ram: Option<Vec<u8>>,
58}
59
60impl Vram {
61    /// Create new VRAM with specified mirroring
62    pub fn new(mirroring: Mirroring) -> Self {
63        Self {
64            nametables: vec![0; 2048], // 2KB
65            palette: vec![0; 32],
66            mirroring,
67            four_screen_ram: if matches!(mirroring, Mirroring::FourScreen) {
68                Some(vec![0; 4096]) // 4KB for four-screen
69            } else {
70                None
71            },
72        }
73    }
74
75    /// Set mirroring mode
76    pub fn set_mirroring(&mut self, mirroring: Mirroring) {
77        self.mirroring = mirroring;
78        if matches!(mirroring, Mirroring::FourScreen) && self.four_screen_ram.is_none() {
79            self.four_screen_ram = Some(vec![0; 4096]);
80        }
81    }
82
83    /// Get current mirroring mode
84    pub fn mirroring(&self) -> Mirroring {
85        self.mirroring
86    }
87
88    /// Read from VRAM address space
89    ///
90    /// Pattern table reads ($0000-$1FFF) should be handled by the mapper.
91    /// This handles nametable ($2000-$2FFF) and palette ($3F00-$3FFF) reads.
92    pub fn read(&self, addr: u16) -> u8 {
93        let addr = addr & 0x3FFF; // Mirror $4000-$FFFF to $0000-$3FFF
94
95        match addr {
96            // Pattern tables - should be handled by mapper
97            // Note: These reads happen during CPU writes to $2007 targeting pattern tables.
98            // Real NES games rarely do this; CHR ROM is accessed by the PPU rendering pipeline
99            // through the mapper, not via the VRAM read() method. Return 0 for now.
100            0x0000..=0x1FFF => {
101                log::trace!("VRAM read from pattern table ${addr:04X} - mapper handles CHR");
102                0
103            }
104
105            // Nametables ($2000-$2FFF) with mirroring
106            0x2000..=0x2FFF => {
107                let mirrored_addr = self.mirror_nametable_addr(addr);
108                self.nametables[mirrored_addr]
109            }
110
111            // Nametable mirror ($3000-$3EFF)
112            0x3000..=0x3EFF => {
113                let mirrored_addr = self.mirror_nametable_addr(addr - 0x1000);
114                self.nametables[mirrored_addr]
115            }
116
117            // Palette RAM ($3F00-$3FFF)
118            0x3F00..=0x3FFF => {
119                let palette_addr = self.mirror_palette_addr(addr);
120                self.palette[palette_addr]
121            }
122
123            _ => unreachable!(),
124        }
125    }
126
127    /// Write to VRAM address space
128    pub fn write(&mut self, addr: u16, value: u8) {
129        let addr = addr & 0x3FFF;
130
131        match addr {
132            // Pattern tables - handled by mapper for CHR-RAM carts
133            // Most games use CHR-ROM (read-only). CHR-RAM writes go through mapper.
134            0x0000..=0x1FFF => {
135                log::trace!("VRAM write to pattern table ${addr:04X} - mapper handles CHR-RAM");
136            }
137
138            // Nametables ($2000-$2FFF) with mirroring
139            0x2000..=0x2FFF => {
140                let mirrored_addr = self.mirror_nametable_addr(addr);
141                self.nametables[mirrored_addr] = value;
142            }
143
144            // Nametable mirror ($3000-$3EFF)
145            0x3000..=0x3EFF => {
146                let mirrored_addr = self.mirror_nametable_addr(addr - 0x1000);
147                self.nametables[mirrored_addr] = value;
148            }
149
150            // Palette RAM ($3F00-$3FFF)
151            0x3F00..=0x3FFF => {
152                let palette_addr = self.mirror_palette_addr(addr);
153                self.palette[palette_addr] = value & 0x3F;
154            }
155
156            _ => unreachable!(),
157        }
158    }
159
160    /// Mirror nametable address according to current mirroring mode
161    fn mirror_nametable_addr(&self, addr: u16) -> usize {
162        let addr = addr & 0x0FFF; // Get offset within nametable space
163        let nametable = (addr / 0x0400) as usize; // Which nametable (0-3)
164        let offset = (addr % 0x0400) as usize; // Offset within nametable
165
166        match self.mirroring {
167            Mirroring::Horizontal => {
168                // A A B B (0=A, 1=A, 2=B, 3=B)
169                let bank = usize::from(nametable >= 2);
170                bank * 0x0400 + offset
171            }
172            Mirroring::Vertical => {
173                // A B A B (0=A, 1=B, 2=A, 3=B)
174                let bank = nametable % 2;
175                bank * 0x0400 + offset
176            }
177            Mirroring::SingleScreenLower => {
178                // A A A A (all map to first 1KB)
179                offset
180            }
181            Mirroring::SingleScreenUpper => {
182                // B B B B (all map to second 1KB)
183                0x0400 + offset
184            }
185            Mirroring::FourScreen => {
186                // A B C D (use external RAM)
187                // This shouldn't be called if four_screen_ram is used
188                nametable * 0x0400 + offset
189            }
190        }
191    }
192
193    /// Mirror palette address according to palette mirroring rules
194    ///
195    /// Palette RAM has special mirroring:
196    /// - $3F10, $3F14, $3F18, $3F1C mirror $3F00, $3F04, $3F08, $3F0C
197    /// - All addresses mirror every 32 bytes
198    fn mirror_palette_addr(&self, addr: u16) -> usize {
199        let mut addr = (addr & 0x1F) as usize; // Mirror to 32 bytes
200
201        // Mirror sprite palette background colors to background palette
202        if addr >= 0x10 && addr.is_multiple_of(4) {
203            addr -= 0x10;
204        }
205
206        addr
207    }
208
209    /// Read palette entry directly (for rendering)
210    #[inline]
211    pub fn read_palette(&self, addr: u8) -> u8 {
212        let addr = self.mirror_palette_addr(0x3F00 | (addr as u16));
213        self.palette[addr]
214    }
215
216    /// Reset VRAM to power-up state
217    pub fn reset(&mut self) {
218        // Clear nametables
219        self.nametables.fill(0);
220        // Clear palette RAM
221        self.palette.fill(0);
222        // Clear four-screen RAM if present
223        if let Some(ref mut ram) = self.four_screen_ram {
224            ram.fill(0);
225        }
226    }
227}
228
229impl Default for Vram {
230    fn default() -> Self {
231        Self::new(Mirroring::Horizontal)
232    }
233}

239    #[test]
240    fn test_horizontal_mirroring() {
241        let vram = Vram::new(Mirroring::Horizontal);
242
243        // Nametables 0 and 1 should map to the same memory
244        assert_eq!(vram.mirror_nametable_addr(0x2000), 0x0000);
245        assert_eq!(vram.mirror_nametable_addr(0x2400), 0x0000);
246
247        // Nametables 2 and 3 should map to the same memory
248        assert_eq!(vram.mirror_nametable_addr(0x2800), 0x0400);
249        assert_eq!(vram.mirror_nametable_addr(0x2C00), 0x0400);
250    }

276    #[test]
277    fn test_palette_mirroring() {
278        let vram = Vram::new(Mirroring::Horizontal);
279
280        // Normal palette addresses
281        assert_eq!(vram.mirror_palette_addr(0x3F00), 0x00);
282        assert_eq!(vram.mirror_palette_addr(0x3F0F), 0x0F);
283
284        // Sprite palette background colors mirror to background palette
285        assert_eq!(vram.mirror_palette_addr(0x3F10), 0x00); // Mirrors $3F00
286        assert_eq!(vram.mirror_palette_addr(0x3F14), 0x04); // Mirrors $3F04
287        assert_eq!(vram.mirror_palette_addr(0x3F18), 0x08); // Mirrors $3F08
288        assert_eq!(vram.mirror_palette_addr(0x3F1C), 0x0C); // Mirrors $3F0C
289
290        // Non-background sprite colors don't mirror
291        assert_eq!(vram.mirror_palette_addr(0x3F11), 0x11);
292        assert_eq!(vram.mirror_palette_addr(0x3F1F), 0x1F);
293    }

349    #[test]
350    fn test_change_mirroring() {
351        let mut vram = Vram::new(Mirroring::Horizontal);
352
353        vram.write(0x2000, 0x42);
354        vram.write(0x2400, 0x55);
355
356        // With horizontal mirroring, both should be 0x55
357        assert_eq!(vram.read(0x2000), 0x55);
358
359        // Change to vertical mirroring
360        vram.set_mirroring(Mirroring::Vertical);
361
362        // Now they should be independent
363        vram.write(0x2000, 0xAA);
364        assert_eq!(vram.read(0x2000), 0xAA);
365        // 0x2400 maps to bank 1 which was never written under vertical mirroring
366        assert_eq!(vram.read(0x2400), 0x00); // Different bank now, not written yet
367    }
