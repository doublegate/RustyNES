1//! Audio resampler for converting APU rate to target sample rate
2//!
3//! The NES APU runs at approximately 1.789773 MHz (NTSC) or 1.662607 MHz (PAL),
4//! but modern audio systems expect 44.1 kHz or 48 kHz. This module provides
5//! high-quality resampling using:
6//!
7//! 1. **Rubato sinc interpolation** - Band-limited resampling for accurate frequency content
8//! 2. **Multi-stage filter chain** - High-pass and low-pass filtering matching NES hardware
9//! 3. **Fallback linear interpolation** - For when rubato isn't available
10//!
11//! # NES Audio Characteristics
12//!
13//! The NES audio path has several filtering stages:
14//! - 90 Hz high-pass filter (removes DC offset)
15//! - 440 Hz high-pass filter (second stage)
16//! - 14 kHz low-pass filter (anti-aliasing before output)
17//!
18//! # Example
19//!
20//! ```rust
21//! use rustynes_apu::resampler::HighQualityResampler;
22//!
23//! let mut resampler = HighQualityResampler::new(48000);
24//!
25//! // Add APU samples (called every CPU cycle at ~1.79 MHz)
26//! for i in 0..10000 {
27//!     let sample = (i as f32 * 0.001).sin();
28//!     resampler.add_sample(sample);
29//! }
30//!
31//! // Retrieve output samples for audio device
32//! let output = resampler.samples();
33//! println!("Generated {} output samples", output.len());
34//!
35//! // Clear buffer after consuming
36//! resampler.clear();
37//! ```
38

129    /// Create a resampler with custom input rate (for PAL systems)
130    ///
131    /// # Arguments
132    ///
133    /// * `output_rate` - Target sample rate (Hz)
134    /// * `input_rate` - APU sample rate (Hz)
135    #[must_use]
136    pub fn with_input_rate(output_rate: u32, input_rate: u32) -> Self {
137        // Use an intermediate rate for two-stage resampling
138        // This helps with the extreme ratio (~1.79MHz -> 48kHz = 37:1)
139        let intermediate_rate = output_rate * 4; // 192 kHz intermediate
140
141        let filter_chain = FilterChain::new(output_rate);
142        let fallback_resampler = LinearResampler::with_input_rate(output_rate, input_rate);
143
144        // Try to create rubato resamplers
145        let (decimator, final_resampler, chunk_size, using_fallback) =
146            Self::create_resamplers(input_rate, intermediate_rate, output_rate);
147
148        // Preallocate work buffers for rubato processing
149        // Stage 1: APU rate (1.79MHz) -> intermediate (192kHz), ~9.3:1 ratio
150        // Stage 2: intermediate (192kHz) -> output (48kHz), 4:1 ratio
151        let stage1_input_cap = chunk_size;
152        let stage1_output_cap = chunk_size / 4; // Conservative estimate
153        let stage2_input_cap = stage1_output_cap;
154        let stage2_output_cap = stage2_input_cap;
155
156        Self {
157            output_rate,
158            input_rate,
159            intermediate_rate,
160            decimator,
161            final_resampler,
162            decimator_input: Vec::with_capacity(chunk_size * 2),
163            intermediate_buffer: Vec::with_capacity(4096),
164            output_buffer: Vec::with_capacity(4096),
165            filter_chain,
166            fallback_resampler,
167            using_fallback,
168            chunk_size,
169            stage1_input_work: Vec::with_capacity(stage1_input_cap),
170            stage1_output_work: Vec::with_capacity(stage1_output_cap),
171            stage2_input_work: Vec::with_capacity(stage2_input_cap),
172            stage2_output_work: Vec::with_capacity(stage2_output_cap),
173        }
174    }
175
176    /// Create rubato resamplers for two-stage decimation
177    #[allow(clippy::type_complexity)]
178    fn create_resamplers(
179        input_rate: u32,
180        intermediate_rate: u32,
181        output_rate: u32,
182    ) -> (
183        Option<Box<FftFixedInOut<f32>>>,
184        Option<Box<FftFixedInOut<f32>>>,
185        usize,
186        bool,
187    ) {
188        // Calculate the decimation ratio
189        let decimation_ratio = f64::from(input_rate) / f64::from(intermediate_rate);
190
191        // Use a reasonable chunk size that works with the ratio
192        // For 1.79MHz -> 192kHz, ratio is ~9.3:1
193        // We need chunk_size * ratio to give integer outputs
194        let chunk_size = 1024;
195
196        // Try to create the decimator (APU rate -> intermediate)
197        let decimator = match FftFixedInOut::<f32>::new(
198            input_rate as usize,
199            intermediate_rate as usize,
200            chunk_size,
201            1,
202        ) {
203            Ok(r) => {
204                debug!(
205                    "Created decimator: {input_rate} Hz -> {intermediate_rate} Hz (ratio {decimation_ratio:.4})"
206                );
207                Some(Box::new(r))
208            }
209            Err(e) => {
210                warn!("Failed to create decimator: {e}, falling back to linear interpolation");
211                return (None, None, chunk_size, true);
212            }
213        };
214
215        // Try to create the final resampler (intermediate -> output)
216        let final_resampler = match FftFixedInOut::<f32>::new(
217            intermediate_rate as usize,
218            output_rate as usize,
219            256,
220            1,
221        ) {
222            Ok(r) => {
223                debug!("Created final resampler: {intermediate_rate} Hz -> {output_rate} Hz");
224                Some(Box::new(r))
225            }
226            Err(e) => {
227                warn!(
228                    "Failed to create final resampler: {e}, falling back to linear interpolation"
229                );
230                return (None, None, chunk_size, true);
231            }
232        };
233
234        (decimator, final_resampler, chunk_size, false)
235    }

237    /// Add a sample from the APU
238    ///
239    /// This should be called every APU cycle (~1.79 MHz).
240    #[inline]
241    pub fn add_sample(&mut self, sample: f32) {
242        if self.using_fallback {
243            // Use linear interpolation fallback
244            self.fallback_resampler.add_sample(sample);
245            return;
246        }
247
248        // Accumulate samples for chunked processing
249        self.decimator_input.push(sample);
250
251        // Process when we have enough samples
252        if self.decimator_input.len() >= self.chunk_size {
253            self.process_chunk();
254        }
255    }
256
257    /// Process a chunk of samples through the resampling stages
258    fn process_chunk(&mut self) {
259        if self.decimator_input.len() < self.chunk_size {
260            return;
261        }
262
263        // Stage 1: Decimate from APU rate to intermediate rate
264        if let Some(ref mut decimator) = self.decimator {
265            let input_frames = decimator.input_frames_next();
266            if self.decimator_input.len() >= input_frames {
267                // Reuse preallocated work buffer instead of allocating new Vec each call
268                self.stage1_input_work.clear();
269                self.stage1_input_work
270                    .extend(self.decimator_input.drain(..input_frames));
271
272                let output_frames = decimator.output_frames_next();
273                self.stage1_output_work.clear();
274                self.stage1_output_work.resize(output_frames, 0.0);
275
276                // rubato needs Vec<Vec<f32>> format - create temporary wrapper slices
277                let input_wrapper = vec![std::mem::take(&mut self.stage1_input_work)];
278                let mut output_wrapper = vec![std::mem::take(&mut self.stage1_output_work)];
279
280                if let Ok((_, _)) =
281                    decimator.process_into_buffer(&input_wrapper, &mut output_wrapper, None)
282                {
283                    self.intermediate_buffer.extend(&output_wrapper[0]);
284                }
285
286                // Restore buffers for reuse (take back ownership)
287                if let Some(buf) = input_wrapper.into_iter().next() {
288                    self.stage1_input_work = buf;
289                }
290                if let Some(buf) = output_wrapper.into_iter().next() {
291                    self.stage1_output_work = buf;
292                }
293            }
294        }
295
296        // Stage 2: Resample from intermediate to output rate
297        if let Some(ref mut final_resampler) = self.final_resampler {
298            let input_frames = final_resampler.input_frames_next();
299            while self.intermediate_buffer.len() >= input_frames {
300                // Reuse preallocated work buffer instead of allocating new Vec each call
301                self.stage2_input_work.clear();
302                self.stage2_input_work
303                    .extend(self.intermediate_buffer.drain(..input_frames));
304
305                let output_frames = final_resampler.output_frames_next();
306                self.stage2_output_work.clear();
307                self.stage2_output_work.resize(output_frames, 0.0);
308
309                // rubato needs Vec<Vec<f32>> format - create temporary wrapper slices
310                let input_wrapper = vec![std::mem::take(&mut self.stage2_input_work)];
311                let mut output_wrapper = vec![std::mem::take(&mut self.stage2_output_work)];
312
313                if let Ok((_, _)) =
314                    final_resampler.process_into_buffer(&input_wrapper, &mut output_wrapper, None)
315                {
316                    // Apply filter chain to output samples
317                    for sample in &output_wrapper[0] {
318                        let filtered = self.filter_chain.process(*sample);
319                        self.output_buffer.push(filtered);
320                    }
321                }
322
323                // Restore buffers for reuse (take back ownership)
324                if let Some(buf) = input_wrapper.into_iter().next() {
325                    self.stage2_input_work = buf;
326                }
327                if let Some(buf) = output_wrapper.into_iter().next() {
328                    self.stage2_output_work = buf;
329                }
330            }
331        }
332    }

395        self.using_fallback
396    }
397
398    /// Flush any remaining samples in the pipeline
399    pub fn flush(&mut self) {
400        // Process any remaining samples with zero-padding if needed
401        while !self.decimator_input.is_empty() {
402            let remaining = self.decimator_input.len();
403            let padding_needed = self.chunk_size.saturating_sub(remaining);
404            self.decimator_input.extend(vec![0.0f32; padding_needed]);
405            self.process_chunk();
406        }
407    }

416/// NES audio filter chain
417///
418/// Implements the analog filter stages present in the NES audio output:
419/// - 90 Hz first-order high-pass (removes DC offset)
420/// - 440 Hz first-order high-pass (shapes bass response)
421/// - 14 kHz first-order low-pass (anti-aliasing)
422pub struct FilterChain {
423    /// First high-pass filter (90 Hz)
424    highpass_1: HighPassFilter,
425    /// Second high-pass filter (440 Hz)
426    highpass_2: HighPassFilter,
427    /// Low-pass filter (14 kHz)
428    lowpass: LowPassFilter,
429    /// Sample rate for filter calculations
430    sample_rate: u32,
431}
432
433impl FilterChain {
434    /// Create a new filter chain for the specified sample rate
435    #[must_use]
436    pub fn new(sample_rate: u32) -> Self {
437        #[allow(clippy::cast_precision_loss)]
438        let rate = sample_rate as f32;
439
440        Self {
441            highpass_1: HighPassFilter::new(HIGHPASS_1_FREQ, rate),
442            highpass_2: HighPassFilter::new(HIGHPASS_2_FREQ, rate),
443            lowpass: LowPassFilter::new(LOWPASS_FREQ, rate),
444            sample_rate,
445        }
446    }
447
448    /// Process a sample through all filter stages
449    #[inline]
450    pub fn process(&mut self, sample: f32) -> f32 {
451        let hp1 = self.highpass_1.process(sample);
452        let hp2 = self.highpass_2.process(hp1);
453        self.lowpass.process(hp2)
454    }
455
456    /// Reset all filters to initial state
457    pub fn reset(&mut self) {
458        self.highpass_1.reset();
459        self.highpass_2.reset();
460        self.lowpass.reset();
461    }
462
463    /// Get the sample rate
464    #[must_use]
465    pub fn sample_rate(&self) -> u32 {
466        self.sample_rate
467    }
468}

470/// First-order high-pass filter (IIR)
471///
472/// Implements: `y[n] = a * (y[n-1] + x[n] - x[n-1])`
473/// where `a = RC / (RC + dt)` and `RC = 1 / (2 * pi * fc)`
474pub struct HighPassFilter {
475    /// Previous input sample
476    prev_input: f32,
477    /// Previous output sample
478    prev_output: f32,
479    /// Filter coefficient
480    alpha: f32,
481}
482
483impl HighPassFilter {
484    /// Create a new high-pass filter
485    ///
486    /// # Arguments
487    ///
488    /// * `cutoff_hz` - Cutoff frequency in Hz
489    /// * `sample_rate` - Sample rate in Hz
490    #[must_use]
491    pub fn new(cutoff_hz: f32, sample_rate: f32) -> Self {
492        let rc = 1.0 / (2.0 * PI * cutoff_hz);
493        let dt = 1.0 / sample_rate;
494        let alpha = rc / (rc + dt);
495
496        Self {
497            prev_input: 0.0,
498            prev_output: 0.0,
499            alpha,
500        }
501    }
502
503    /// Process a sample through the filter
504    #[inline]
505    pub fn process(&mut self, input: f32) -> f32 {
506        let output = self.alpha * (self.prev_output + input - self.prev_input);
507        self.prev_input = input;
508        self.prev_output = output;
509        output
510    }
511
512    /// Reset filter state
513    pub fn reset(&mut self) {
514        self.prev_input = 0.0;
515        self.prev_output = 0.0;
516    }
517}

519/// First-order low-pass filter (IIR)
520///
521/// Implements: `y[n] = y[n-1] + a * (x[n] - y[n-1])`
522/// where `a = dt / (RC + dt)` and `RC = 1 / (2 * pi * fc)`
523pub struct LowPassFilter {
524    /// Previous output sample
525    prev_sample: f32,
526    /// Filter coefficient (0.0-1.0)
527    alpha: f32,
528}
529
530impl LowPassFilter {
531    /// Create a new low-pass filter
532    ///
533    /// # Arguments
534    ///
535    /// * `cutoff_hz` - Cutoff frequency in Hz
536    /// * `sample_rate` - Sample rate in Hz
537    #[must_use]
538    pub fn new(cutoff_hz: f32, sample_rate: f32) -> Self {
539        let rc = 1.0 / (2.0 * PI * cutoff_hz);
540        let dt = 1.0 / sample_rate;
541        let alpha = dt / (rc + dt);
542
543        Self {
544            prev_sample: 0.0,
545            alpha,
546        }
547    }
548
549    /// Process a sample through the filter
550    #[inline]
551    pub fn process(&mut self, sample: f32) -> f32 {
552        let output = self.prev_sample + self.alpha * (sample - self.prev_sample);
553        self.prev_sample = output;
554        output
555    }
556
557    /// Reset filter state
558    pub fn reset(&mut self) {
559        self.prev_sample = 0.0;
560    }
561}

563/// Simple linear interpolation resampler (fallback)
564///
565/// Used when rubato cannot be initialized or for simpler use cases.
566pub struct LinearResampler {
567    /// Target output sample rate
568    output_rate: u32,
569    /// APU sample rate
570    input_rate: u32,
571    /// Fractional time accumulator
572    time_accumulator: f32,
573    /// Previous sample for interpolation
574    prev_sample: f32,
575    /// Output sample buffer
576    buffer: Vec<f32>,
577}
578
579impl LinearResampler {
580    /// Create a new linear resampler
581    #[must_use]
582    pub fn new(output_rate: u32) -> Self {
583        Self::with_input_rate(output_rate, APU_RATE_NTSC)
584    }
585
586    /// Create a resampler with custom input rate
587    #[must_use]
588    pub fn with_input_rate(output_rate: u32, input_rate: u32) -> Self {
589        Self {
590            output_rate,
591            input_rate,
592            time_accumulator: 0.0,
593            prev_sample: 0.0,
594            buffer: Vec::with_capacity(2048),
595        }
596    }
597
598    /// Add a sample from the APU
599    pub fn add_sample(&mut self, sample: f32) {
600        #[allow(clippy::cast_precision_loss)]
601        let time_step = self.output_rate as f32 / self.input_rate as f32;
602        self.time_accumulator += time_step;
603
604        while self.time_accumulator >= 1.0 {
605            let t = self.time_accumulator - 1.0;
606            let output = self.prev_sample + (sample - self.prev_sample) * t;
607            self.buffer.push(output);
608            self.time_accumulator -= 1.0;
609        }
610
611        self.prev_sample = sample;
612    }
613
614    /// Get reference to output samples
615    #[must_use]
616    pub fn samples(&self) -> &[f32] {
617        &self.buffer
618    }
619
620    /// Clear the output buffer
621    pub fn clear(&mut self) {
622        self.buffer.clear();
623    }
624
625    /// Check if resampler has at least `min_samples` ready
626    #[must_use]
627    pub fn is_ready(&self, min_samples: usize) -> bool {
628        self.buffer.len() >= min_samples
629    }
630
631    /// Get the current number of samples
632    #[must_use]
633    pub fn len(&self) -> usize {
634        self.buffer.len()
635    }
636
637    /// Check if buffer is empty
638    #[must_use]
639    pub fn is_empty(&self) -> bool {
640        self.buffer.is_empty()
641    }
642
643    /// Reset the resampler state
644    pub fn reset(&mut self) {
645        self.buffer.clear();
646        self.time_accumulator = 0.0;
647        self.prev_sample = 0.0;
648    }
649
650    /// Get output sample rate
651    #[must_use]
652    pub fn output_rate(&self) -> u32 {
653        self.output_rate
654    }
655
656    /// Get input sample rate
657    #[must_use]
658    pub fn input_rate(&self) -> u32 {
659        self.input_rate
660    }
661}
662
663impl Default for LinearResampler {
664    fn default() -> Self {
665        Self::new(SAMPLE_RATE_48000)
666    }
667}
